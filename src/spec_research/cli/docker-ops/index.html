    2. 服务管理¶

    # 查看服务状态
sudo systemctl status docker
sudo systemctl status containerd

# 启动 / 停止 / 重启
sudo systemctl start docker
sudo systemctl stop docker
sudo systemctl restart docker

# 开机自启
sudo systemctl enable docker

# 查看服务配置
systemctl cat docker

    Socket 权限¶

    Docker socket 位于 /var/run/docker.sock，权限为 660，所属组为 docker。当前用户 lion 已在 docker 组中，可直接使用 docker 命令无需 sudo。

    # 确认当前用户在 docker 组
groups lion

# 将新用户加入 docker 组（免 sudo）
sudo usermod -aG docker <username>
newgrp docker  # 当前会话生效

    3. Daemon 配置¶

    配置文件: /etc/docker/daemon.json

    {
  "registry-mirrors": [
    "https://docker.m.daocloud.io"
  ]
}

    常用配置项¶

    {
  "registry-mirrors": ["https://your-mirror.com"],
  "storage-driver": "overlayfs",
  "log-driver": "json-file",
  "log-opts": {
    "max-size": "10m",
    "max-file": "3"
  },
  "insecure-registries": ["192.168.1.100:5000"],
  "live-restore": true
}

    修改配置后需重启:

    sudo systemctl daemon-reload
sudo systemctl restart docker

    常用操作¶

    # 拉取镜像
docker pull nginx:alpine

# 列出镜像
docker images
docker images -a          # 包含中间层
docker images -f dangling=true  # 查看悬空镜像

# 删除镜像
docker rmi <image_id>
docker image prune        # 删除悬空镜像
docker image prune -a     # 删除所有未使用的镜像

# 构建镜像
docker build -t myapp:v1 .
docker buildx build --platform linux/amd64,linux/arm64 -t myapp:latest .

# 导出/导入
docker save -o app.tar myapp:v1
docker load -i app.tar

# 查看镜像历史
docker history <image_id>

# 查看镜像详情
docker inspect <image_id>

# 查看容器详情
docker inspect <name|id>

    9. 磁盘清理¶

    当前 Docker 总磁盘占用约 29.2 GB，其中大部分可回收:

    类型 总计 可回收
    镜像 20.76GB 13.04GB (62%)
    容器 2.13GB 2.13GB (100%)
    卷 3.03GB 3.02GB (99%)
    构建缓存 3.29GB 1.32GB 
    # 查看磁盘使用
docker system df
docker system df -v         # 详细信息

# 一键清理所有未使用资源 (危险, 会删除停止的容器、未使用网络、悬空镜像、构建缓存)
docker system prune

# 更彻底的清理 (包含未使用的镜像)
docker system prune -a

# 清理所有 (包含未使用的卷)
docker system prune -a --volumes

# 单独清理
docker container prune      # 停止的容器
docker image prune          # 悬空镜像
docker image prune -a       # 所有未使用的镜像
docker volume prune         # 未使用的卷
docker builder prune        # 构建缓存

    11. Buildx 多架构构建¶

    # 查看构建器
docker buildx ls

# 创建多架构构建器
docker buildx create --name mybuilder --use
docker buildx inspect --bootstrap

# 构建并推送多平台镜像
docker buildx build --platform linux/amd64,linux/arm64 \
  -t myregistry/myapp:latest --push .

# 仅本地构建
docker buildx build --load -t myapp:latest .

    12. 日志管理¶

    当前日志驱动为 json-file，建议配置日志轮转防止磁盘爆满。

    在 /etc/docker/daemon.json 中添加:

    {
  "log-driver": "json-file",
  "log-opts": {
    "max-size": "10m",
    "max-file": "3"
  }
}

    # 查看容器日志
docker logs <container>
docker logs -f --since 1h <container>
docker logs --tail 200 <container>

    13. 常见故障排查¶

    # Docker 无法启动
sudo journalctl -u docker --no-pager -n 50

# 容器反复退出
docker inspect <container> --format '{{.State.ExitCode}}'
docker logs <container>

# 磁盘空间不足
df -h /var/lib/docker
docker system df

# 网络不通
docker network inspect <network>
docker exec -it <container> ping <target>

# 权限问题
ls -la /var/run/docker.sock
groups  # 确认在 docker 组中
