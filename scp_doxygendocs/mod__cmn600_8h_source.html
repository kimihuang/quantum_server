56/**
57 * @brief 内存区域配置类型
58 *
59 * @details 定义 CMN-600 互连中内存区域的类型，
60 * 不同类型决定了区域由哪类节点服务和如何进行地址映射。
61 */
62enum mod_cmn600_memory_region_type {
63 /** I/O 区域，由专用的 HN-I 或 HN-D 节点服务 */
64 MOD_CMN600_MEMORY_REGION_TYPE_IO,
65
66 /**
67 * 系统缓存区域，由系统中所有 HN-F 节点服务
68 * （使用 HN-F 节点池进行一致性缓存）
69 */
70 MOD_CMN600_MEMORY_REGION_TYPE_SYSCACHE,
71
72 /**
73 * 系统缓存的非哈希子区域，由 HN-F 节点服务
74 * （不进行哈希映射，直接访问指定目标）
75 */
76 MOD_CMN600_REGION_TYPE_SYSCACHE_SUB,
77
78 /**
79 * CCIX 访问区域，用于跨片 CCIX 一致性通信
80 */
81 MOD_CMN600_REGION_TYPE_CCIX,
82
83 /**
84 * 系统缓存分组中的非哈希区域
85 */
86 MOD_CMN600_REGION_TYPE_SYSCACHE_NONHASH,
87};
88
89/**
90 * @brief 内存区域映射描述符
91 *
92 * @details 描述一个内存区域的基地址、大小、类型和目标节点，
93 * 用于配置 RN-SAM（请求节点系统地址映射）寄存器。
94 */
95struct mod_cmn600_memory_region_map {
96 /** @brief 区域基地址，须按 SAM_GRANULARITY（64MB）对齐 */
97 uint64_t base;
98
99 /** @brief 区域大小（字节），须为 SAM_GRANULARITY 的 2 的幂次倍 */
100 uint64_t size;
101
102 /** @brief 区域配置类型，参见 enum mod_cmn600_memory_region_type */
103 enum mod_cmn600_memory_region_type type;
104
105 /**
106 * @brief 目标节点标识符
107 *
108 * @note 对于 MOD_CMN600_MEMORY_REGION_TYPE_SYSCACHE 类型区域不使用此字段，
109 * 因为系统缓存区域使用系统中所有可用的 HN-F 节点池。
110 */
111 unsigned int node_id;
112};
113
114/**
115 * @brief CMN-600 模块配置数据
116 *
117 * @details 包含 CMN-600 互连的完整硬件配置信息，
118 * 由平台定义并通过 SCP 框架在模块初始化时传入。
119 * 该结构体定义了互连 mesh 拓扑、节点映射和时钟依赖等。
120 */
121struct mod_cmn600_config {
122 /** @brief CMN-600 外设基地址，即 MMIO 映射的起始地址 */
123 uintptr_t base;
124
125 /** @brief 互连 mesh 的 X 轴尺寸，有效范围 1-8 */
126 unsigned int mesh_size_x;
127
128 /** @brief 互连 mesh 的 Y 轴尺寸，有效范围 1-8 */
129 unsigned int mesh_size_y;
130
131 /** @brief 包含全局配置的默认 HN-D 节点标识符，用于定位根节点（Configuration Manager） */
132 unsigned int hnd_node_id;
133
134 /**
135 * @brief HN-F 节点的目标 SN-F 节点映射表
136 *
137 * @details 表中每个条目对应一个 HN-F 节点，
138 * 以 HN-F 的逻辑 ID 作为索引，值为该 HN-F 对应的 SN-F 节点 ID。
139 */
140 const unsigned int *snf_table;
141
142 /** @brief snf_table 中的条目数量，须等于系统中 HN-F 的总数 */
143 size_t snf_count;
144
145 /** @brief 主节点的 SA（Slave Agent）数量 */
146 unsigned int sa_count;
147
148 /** @brief 内存区域映射表，描述所有需配置的内存区域 */
149 const struct mod_cmn600_memory_region_map *mmap_table;
150
151 /** @brief mmap_table 中的条目数量 */
152 size_t mmap_count;
153
154 /** @brief 芯片的地址空间大小，用于多芯片场景下的地址偏移计算 */
155 uint64_t chip_addr_space;
156
157 /** @brief 该设备依赖的时钟标识符，若为 FWK_ID_NONE 则不等待时钟 */
158 fwk_id_t clock_id;
159
160 /**
161 * @brief HN-F CAL 支持使能标志
162 *
163 * @details 当设为 true 时，使能 HN-F 的 CAL (Coherent Address Lookup) 模式。
164 * 此标志仅当 HN-F 连接到 CAL 端口时生效
165 * （连接到 CAL 端口时，HN-F 的节点 ID 为奇数）。
166 *
167 * @note CAL 模式仅 CMN-600 r2p0 及以上版本支持
168 */
169 bool hnf_cal_mode;
170};
171
172/**
173 * @brief CCIX HA 内存映射表条目

178struct mod_cmn600_ccix_ha_mmap {

180 /** @brief Home Agent 节点 ID */

182
183 /** @brief 区域基地址 */
184 uint64_t base;
185
186 /** @brief 区域大小（字节） */
187 uint64_t size;
188};

191/**
192 * @brief CMN-600 远端节点 CCIX 配置数据
193 *
194 * @details 由远端芯片通过 SCMI 协议传递的 CCIX 配置信息，
195 * 用于配置本地 CCIX Gateway 以建立跨片一致性链路。
196 */
197struct mod_cmn600_ccix_remote_node_config {
198 /**
199 * @brief 远端缓存代理 (RN-F) 数量

204
205 /** @brief 远端 SA（Slave Agent）数量 */
206 uint8_t remote_sa_count;
207
208 /** @brief 远端 HA（Home Agent）数量 */
209 uint8_t remote_ha_count;
210
211 /** @brief CCIX 虚通道使用的 PCIe Traffic Class */
212 uint8_t ccix_tc;
213
214 /** @brief CCIX 消息打包使能标志 */
215 bool ccix_msg_pack_enable;
216
217 /** @brief CCIX 链路使能的 PCIe 总线号 */
218 uint8_t pcie_bus_num;
219
220 /** @brief CCIX 链路标识符，有效值为 0-2 */
221 uint8_t ccix_link_id;
222
223 /** @brief CCIX 优化 TLP 模式使能标志 */
224 bool ccix_opt_tlp;
225
226 /** @brief 远端 HA 内存映射表条目数量 */
227 uint8_t remote_ha_mmap_count;
228
229 /** @brief SMP 模式使能标志 */
230 bool smp_mode;
231
232 /** @brief 远端 HA 内存映射表，最多 MAX_HA_MMAP_ENTRIES 个条目 */
233 struct mod_cmn600_ccix_ha_mmap remote_ha_mmap[MAX_HA_MMAP_ENTRIES];
234
235 /** @brief CCIX 最大包大小编码，参见 CXLA_CCIX_PROP_MAX_PACK_SIZE_xxx */
236 uint8_t ccix_max_packet_size;
237};

261 /** @brief 主节点 CCIX 最大 Memory Request 发送信用数 */

266
267 /** @brief 主节点 CCIX 最大 Data 发送信用数 */
268 uint16_t ccix_data_credits;
269
270 /** @brief 主节点 CCIX 最大包大小编码 */
271 uint8_t ccix_max_packet_size;
272
273 /** @brief 主节点 CCIX 优化 TLP 模式能力 */
274 bool ccix_opt_tlp;
275
276 /** @brief 主节点 CCIX 消息打包能力 */
277 bool ccix_msg_pack_enable;

280/**
281 * @brief CMN-600 CCIX 配置 API
282 *
283 * @details 定义 CCIX 配置接口，供 SCMI 等上层模块调用，
284 * 实现跨片一致性互连的配置流程。
285 */
286struct mod_cmn600_ccix_config_api {
287 /**
288 * @brief 获取 CCIX 主节点配置
289 *
290 * @param[out] config CCIX 主节点配置数据
291 *
292 * @retval FWK_SUCCESS 操作成功
293 * @return 其他错误码表示操作失败
294 */
295 int (*get_config)(struct mod_cmn600_ccix_host_node_config *config);
296
297 /**
298 * @brief 设置 CCIX 远端节点配置
299 *
300 * @param[in] config CCIX 远端节点配置数据
301 *
302 * @retval FWK_SUCCESS 操作成功
303 * @return 其他错误码表示操作失败
304 */
305 int (*set_config)(struct mod_cmn600_ccix_remote_node_config *config);
306
307 /**
308 * @brief 触发 CCIX 协议信用交换
309 *
310 * @param[in] link_id 执行协议信用交换的链路标识符
311 *
312 * @retval FWK_SUCCESS 操作成功
313 * @return 其他错误码表示操作失败
314 */
315 int (*exchange_protocol_credit)(uint8_t link_id);
316
317 /**
318 * @brief 配置进入系统一致性域
319 *
320 * @param[in] link_id 使能一致性的链路标识符
321 *
322 * @retval FWK_SUCCESS 操作成功
323 * @return 其他错误码表示操作失败
324 */
325 int (*enter_system_coherency)(uint8_t link_id);
326
327 /**
328 * @brief 进入 DVM (Distributed Virtual Memory) 域
329 *
330 * @param[in] link_id 使能 DVM 域的链路标识符
331 *
332 * @retval FWK_SUCCESS 操作成功
333 * @return 其他错误码表示操作失败
334 */
335 int (*enter_dvm_domain)(uint8_t link_id);
336};
