79 * @brief 根据父节点和子节点索引获取子节点指针
80 */
81void *get_child_node(uintptr_t base, void *node_base, unsigned int child_index)
82{
83 struct node_header *node = node_base;
84 uint32_t child_pointer;
85 unsigned int offset;
86 void *child_node;
87
88 child_pointer = node->CHILD_POINTER[child_index];
89 offset = child_pointer & CMN600_CHILD_POINTER_OFFSET;
90
91 child_node = (void *)(base + offset);
92 return child_node;

95/**
96 * @brief 通过父节点的子节点指针获取子节点的物理标识符
97 *
98 * @details 从子节点指针中提取节点 ID，将内部编码转换为标准的
99 * 节点 ID 格式（Y坐标|X坐标|端口号）。
100 */
101unsigned int get_child_node_id(void *node_base,
102 unsigned int child_index)
103{
104 struct node_header *node = node_base;
105 uint32_t node_pointer;
106 unsigned int device_id;
107
108 node_pointer = (node->CHILD_POINTER[child_index] &
109 CMN600_CHILD_POINTER_EXT_NODE_POINTER) >>
110 CMN600_CHILD_POINTER_EXT_NODE_POINTER_POS;
111
112 /* 将内部编码重组为标准节点 ID 格式 */
113 device_id = (((node_pointer >> 6) & 0xff) << 3) |
114 ((node_pointer & 0x1) << 2) |
115 ((node_pointer >> 2) & 0x3);
116
117 return device_id;
118}

187/**
188 * @brief 将内存区域大小转换为 CMN-600 寄存器使用的大小编码
189 *
190 * @details 编码公式：n = log2(size / SAM_GRANULARITY)
191 */
192uint64_t sam_encode_region_size(uint64_t size)
193{
194 uint64_t blocks;
195 uint64_t result;
196
197 /* 大小必须是 SAM_GRANULARITY 的倍数 */
198 fwk_assert((size % SAM_GRANULARITY) == 0);
199
200 /* 大小必须是 2 的幂次 */
201 fwk_assert((size & (size - 1)) == 0);
202
203 blocks = size / SAM_GRANULARITY;
204 result = fwk_math_log2(blocks);
205
206 return result;
207}

209/**
210 * @brief 配置一个内存区域条目
211 *
212 * @details 将有效位、目标节点类型、大小编码和基地址编码
213 * 写入 RN-SAM 区域描述符寄存器的指定位偏移位置。
214 */
215void configure_region(volatile uint64_t *reg, unsigned int bit_offset,
216 uint64_t base, uint64_t size, enum sam_node_type node_type)
217{
218 uint64_t value;
219
220 fwk_assert(reg != NULL);
221 fwk_assert((base % size) == 0);
222
223 value = CMN600_RNSAM_REGION_ENTRY_VALID;
224 value |= node_type << CMN600_RNSAM_REGION_ENTRY_TYPE_POS;
225 value |= sam_encode_region_size(size) << CMN600_RNSAM_REGION_ENTRY_SIZE_POS;
226 value |= (base / SAM_GRANULARITY) << CMN600_RNSAM_REGION_ENTRY_BASE_POS;
227
228 *reg &= ~(CMN600_RNSAM_REGION_ENTRY_MASK << bit_offset);
229 *reg |= value << bit_offset;
230}

300/**
301 * @brief 根据外设基地址、HN-D 节点 ID 和 mesh 尺寸获取根节点描述符
302 *
303 * @details 根据 mesh 尺寸确定坐标编码位数，从 HN-D 节点 ID 中
304 * 提取坐标信息，计算根节点（Configuration Manager）的地址偏移。
305 * 该函数同时设置全局变量 encoding_bits 和 mask_bits，
306 * 供 get_node_pos_x() 和 get_node_pos_y() 使用。
307 */
308struct cmn600_cfgm_reg *get_root_node(uintptr_t base, unsigned int hnd_node_id,
309 unsigned int mesh_size_x, unsigned int mesh_size_y)
310{
311 unsigned int node_pos_x;
312 unsigned int node_pos_y;
313 unsigned int node_port;
314 uintptr_t offset;
315
316 /*
317 * 根据 CMN-600 规范，根据 mesh 尺寸确定坐标编码位数：
318 * mesh 任一维度超过 4 时使用 3 位编码，否则使用 2 位。
319 */
320 encoding_bits = ((mesh_size_x > 4) || (mesh_size_y > 4)) ? 3 : 2;
321
322 /* 从节点标识符中提取坐标 */
323 mask_bits = (1 << encoding_bits) - 1;
324 node_pos_y = (hnd_node_id >> CMN600_NODE_ID_Y_POS) & mask_bits;
325 node_pos_x = (hnd_node_id >> (CMN600_NODE_ID_Y_POS + encoding_bits)) &
326 mask_bits;
327 node_port = (hnd_node_id >> CMN600_NODE_ID_PORT_POS) &
328 CMN600_NODE_ID_PORT_MASK;
329
330 /* 计算节点地址偏移 */
331 offset = (node_pos_y << CMN600_ROOT_NODE_OFFSET_Y_POS) |
332 (node_pos_x << (CMN600_ROOT_NODE_OFFSET_Y_POS + encoding_bits)) |
333 (node_port << CMN600_ROOT_NODE_OFFSET_PORT_POS);
334
335 return (struct cmn600_cfgm_reg *)(base + offset);
336}
