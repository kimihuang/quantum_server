26/**

41/**

55/**

347/**
348 * @brief CCIX 链路 UP 等待条件枚举
349 *
350 * @details 定义 CCIX 链路建立过程中各阶段需要轮询的寄存器条件，
351 * 用于 timer_api->wait() 的条件回调函数。
352 * 链路建立序列按此枚举值的顺序依次执行。
353 */
354enum cxg_link_up_wait_cond {
355 CXG_LINK_CTRL_EN_BIT_SET, /**< 等待 CXRA/CXHA 链路控制 EN 位置位 */
356 CXG_LINK_CTRL_UP_BIT_CLR, /**< 等待 CXRA/CXHA 链路控制 UP 位清零 */
357 CXG_LINK_STATUS_DWN_BIT_SET, /**< 等待 CXRA/CXHA 链路状态 DOWN 位置位 */
358 CXG_LINK_STATUS_DWN_BIT_CLR, /**< 等待 CXRA/CXHA 链路状态 DOWN 位清零 */
359 CXG_LINK_STATUS_ACK_BIT_SET, /**< 等待 CXRA/CXHA 链路状态 ACK 位置位 */
360 CXG_LINK_STATUS_ACK_BIT_CLR, /**< 等待 CXRA/CXHA 链路状态 ACK 位清零 */
361 CXG_LINK_STATUS_HA_DVMDOMAIN_ACK_BIT_SET, /**< 等待 CXHA 链路状态 DVMDOMAIN ACK 位置位 */
362 CXG_LINK_STATUS_RA_DVMDOMAIN_ACK_BIT_SET, /**< 等待 CXRA 链路状态 DVMDOMAIN ACK 位置位 */
363 CXG_LINK_UP_SEQ_COUNT, /**< 链路建立序列总步骤数 */
364};
