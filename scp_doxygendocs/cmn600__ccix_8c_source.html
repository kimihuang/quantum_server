44/**
45 * @brief CCIX 链路等待条件回调函数
46 *
47 * @details 根据 wait_data->cond 检查 CXRA/CXHA 链路寄存器的状态，
48 * 判断指定的链路条件是否满足。该函数作为回调传递给
49 * timer_api->wait()，用于轮询链路状态变化。
50 *
51 * @param[in] data 指向 struct cxg_wait_condition_data 的指针
52 *
53 * @return true 指定条件已满足；false 条件未满足
54 */
55static bool cxg_link_wait_condition(void *data)
56{
57 uint64_t val1;
58 uint64_t val2;
59 uint8_t link_id;
60
61 fwk_assert(data != NULL);
62
63 struct cxg_wait_condition_data *wait_data =
64 (struct cxg_wait_condition_data *)data;
65 struct cmn600_ctx *ctx = wait_data->ctx;
66 link_id = wait_data->link_id;
67
68 switch (wait_data->cond) {
69 case CXG_LINK_CTRL_EN_BIT_SET:
70 val1 = ctx->cxg_ra_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_CTRL;
71 val2 = ctx->cxg_ha_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_CTRL;
72 return (((val1 & CXG_LINK_CTRL_EN_MASK) &&
73 (val2 & CXG_LINK_CTRL_EN_MASK)) != 0);
74
75 case CXG_LINK_CTRL_UP_BIT_CLR:
76 val1 = ctx->cxg_ra_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_CTRL;
77 val2 = ctx->cxg_ha_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_CTRL;
78 return ((((val1 & CXG_LINK_CTRL_UP_MASK) == 0) &&
79 ((val2 & CXG_LINK_CTRL_UP_MASK) == 0)) != 0);
80
81 case CXG_LINK_STATUS_DWN_BIT_SET:
82 val1 = ctx->cxg_ra_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_STATUS;
83 val2 = ctx->cxg_ha_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_STATUS;
84 return (((val1 & CXG_LINK_STATUS_DOWN_MASK) &&
85 (val2 & CXG_LINK_STATUS_DOWN_MASK)) != 0);
86
87 case CXG_LINK_STATUS_DWN_BIT_CLR:
88 val1 = ctx->cxg_ra_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_STATUS;
89 val2 = ctx->cxg_ha_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_STATUS;
90 return ((((val1 & CXG_LINK_STATUS_DOWN_MASK) == 0) &&
91 ((val2 & CXG_LINK_STATUS_DOWN_MASK) == 0)) != 0);
92
93 case CXG_LINK_STATUS_ACK_BIT_SET:
94 val1 = ctx->cxg_ra_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_STATUS;
95 val2 = ctx->cxg_ha_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_STATUS;
96 return (((val1 & CXG_LINK_STATUS_ACK_MASK) &&
97 (val2 & CXG_LINK_STATUS_ACK_MASK)) != 0);
98
99 case CXG_LINK_STATUS_ACK_BIT_CLR:
100 val1 = ctx->cxg_ra_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_STATUS;
101 val2 = ctx->cxg_ha_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_STATUS;
102 return ((((val1 & CXG_LINK_STATUS_ACK_MASK) == 0) &&
103 ((val2 & CXG_LINK_STATUS_ACK_MASK) == 0)) != 0);
104
105 case CXG_LINK_STATUS_HA_DVMDOMAIN_ACK_BIT_SET:
106 val1 = ctx->cxg_ha_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_STATUS;
107 return (((val1 & CXG_LINK_STATUS_DVMDOMAIN_ACK_MASK)) != 0);
108
109 case CXG_LINK_STATUS_RA_DVMDOMAIN_ACK_BIT_SET:
110 val1 = ctx->cxg_ra_reg->LINK_REGS[link_id].CXG_PRTCL_LINK_STATUS;
111 return (((val1 & CXG_LINK_STATUS_DVMDOMAIN_ACK_MASK)) != 0);
112
113 default:
114 fwk_unexpected();
115 return false;
116 }
117}

119/**
120 * @brief 使能 SMP (Symmetric Multiprocessing) 模式
121 *
122 * @details 根据 CMN-600 修订版本采用不同的 SMP 模式配置方式：
123 * - r2p0：通过 CXRA/CXHA UNIT_INFO 的只读位判断硬件是否支持
124 * - r3p0+：通过 CXRA/CXHA/CXLA AUX_CTRL 的可写位配置使能
125 * - r1p3 及以下：无 SMP 模式寄存器，直接返回成功
126 *
127 * @param[in] ctx CMN-600 模块上下文指针
128 *
129 * @return FWK_SUCCESS 成功；FWK_E_SUPPORT r2p0 版本硬件不支持 SMP 模式
130 */
131static int enable_smp_mode(struct cmn600_ctx *ctx)
132{
133 if (get_cmn600_revision(ctx->root) == CMN600_PERIPH_ID_2_REV_R2_P0) {
134 /*
135 * CMN-600 r2p0 通过 CXRA/CXHA 寄存器的只读位判断
136 * SMP 模式是否在硬件中使能。
137 */
138 if (((ctx->cxg_ra_reg->CXG_RA_UNIT_INFO &
139 CXG_RA_UNIT_INFO_SMP_MODE_RO_MASK) !=
140 CXG_RA_UNIT_INFO_SMP_MODE_RO_MASK) ||
141 ((ctx->cxg_ha_reg->CXG_HA_UNIT_INFO &
142 CXG_HA_UNIT_INFO_SMP_MODE_RO_MASK) !=
143 CXG_HA_UNIT_INFO_SMP_MODE_RO_MASK)) {
144 FWK_LOG_INFO(MOD_NAME "SMP Mode not supported");
145 return FWK_E_SUPPORT;
146 }
147
148 FWK_LOG_INFO(MOD_NAME "SMP Mode supported");
149 return FWK_SUCCESS;
150 } else if (get_cmn600_revision(ctx->root) >= CMN600_PERIPH_ID_2_REV_R3_P0) {
151 /*
152 * CMN-600 r3p0 及以上版本允许软件配置 SMP 模式
153 */
154 ctx->cxg_ra_reg->CXG_RA_AUX_CTRL |=
155 (1 << CXG_RA_AUX_CTRL_SMP_MODE_RW_SHIFT_VAL);
156 ctx->cxg_ha_reg->CXG_HA_AUX_CTRL |=
157 (1 << CXG_HA_AUX_CTRL_SMP_MODE_RW_SHIFT_VAL);
158 ctx->cxla_reg->CXLA_AUX_CTRL |=
159 ((uint64_t)0x1 << CXLA_AUX_CTRL_SMP_MODE_SHIFT_VAL);
160 FWK_LOG_INFO(MOD_NAME "SMP MODE enabled");
161 return FWK_SUCCESS;
162 }
163
164 /*
165 * CMN-600 r1p3 及以下版本不提供读取或配置 SMP 模式的寄存器
166 */
167 return FWK_SUCCESS;
168}

170/**
171 * @brief 编程 CXRA RN-F LDID-to-RAID 映射寄存器
172 *
173 * @details 将 RN-F 的 LDID 值映射到对应的 RAID，
174 * 并设置有效位。每个 64 位寄存器容纳 8 个映射项。
175 *
176 * @param[in] ctx CMN-600 模块上下文指针
177 * @param[in] ldid_value LDID 值
178 * @param[in] raid 分配的 RAID 值
179 *
180 * @note 调用后 ctx->raid_value 会递增
181 */
182static void program_cxg_ra_rnf_ldid_to_raid_reg(struct cmn600_ctx *ctx,
183 uint8_t ldid_value, uint8_t raid)
184{
185 uint32_t reg_offset = 0;
186 uint32_t ldid_value_offset = 0;
187
188 /* 每个 64 位 RA RNF LDID-to-RAID 寄存器容纳 8 个 LDID */
189 reg_offset = ldid_value / 8;
190 ldid_value_offset = ldid_value % 8;
191
192 /* 写入 RAID 值到 LDID-to-RAID 寄存器 */
193 ctx->cxg_ra_reg->CXG_RA_RNF_LDID_TO_RAID_REG[reg_offset] |=
194 ((uint64_t)raid <<
195 (ldid_value_offset * NUM_BITS_RESERVED_FOR_RAID));
196
197 /* 设置对应的有效位 */
198 ctx->cxg_ra_reg->CXG_RA_RNF_LDID_TO_RAID_VAL |=
199 ((uint64_t)0x1 << ldid_value);
200
201 /* 递增 raid_value */
202 ctx->raid_value++;
203}

276 * @brief 编程 AgentID-to-LinkID 映射寄存器
277 *
278 * @details 在 CXRA、CXHA 和 CXLA 的 AgentID-to-LinkID 寄存器中
279 * 写入映射关系，并设置对应的有效位。
280 * 每个 64 位寄存器容纳 8 个映射项（每项 8 位）。
281 *
282 * @param[in] ctx CMN-600 模块上下文指针
283 * @param[in] agent_id 代理标识符
284 * @param[in] link_id CCIX 链路标识符

286static void program_agentid_to_linkid_reg(struct cmn600_ctx *ctx,
287 uint8_t agent_id, uint8_t link_id)
288{
289 uint32_t reg_index = 0;
290 uint32_t agentid_bit_offset = 0;
291
292 /* 每个 64 位寄存器容纳 8 个 AgentID/LinkID 映射 */
293 reg_index = agent_id / 8;
294 agentid_bit_offset = agent_id % 8;
295
296 /* 写入 AgentID-to-LinkID 映射 */
297 ctx->cxg_ra_reg->CXG_RA_AGENTID_TO_LINKID_REG[reg_index] |=
298 ((uint64_t)link_id <<
299 (agentid_bit_offset * NUM_BITS_RESERVED_FOR_LINK_ID));
300 ctx->cxg_ha_reg->CXG_HA_AGENTID_TO_LINKID_REG[reg_index] |=
301 ((uint64_t)link_id <<
302 (agentid_bit_offset * NUM_BITS_RESERVED_FOR_LINK_ID));
303 ctx->cxla_reg->CXLA_AGENTID_TO_LINKID_REG[reg_index] |=
304 ((uint64_t)link_id <<
305 (agentid_bit_offset * NUM_BITS_RESERVED_FOR_LINK_ID));
306
307 /* 设置对应的有效位 */
308 ctx->cxg_ra_reg->CXG_RA_AGENTID_TO_LINKID_VAL |=
309 ((uint64_t)0x1 << agent_id);
310 ctx->cxg_ha_reg->CXG_HA_AGENTID_TO_LINKID_VAL |=
311 ((uint64_t)0x1 << agent_id);
312 ctx->cxla_reg->CXLA_AGENTID_TO_LINKID_VAL |= ((uint64_t)0x1 << agent_id);
313}

364 * @brief 编程 HN-F LDID-to-CHI-Node-ID 寄存器
365 *
366 * @details 在所有 HN-F 节点的 RN_PHYS_ID 寄存器中写入
367 * 远端 RN-F 代理的节点 ID、本地/远端标志和有效位。
368 * 每个 64 位寄存器容纳 2 个配置项（每项 32 位）。
369 *
370 * @param[in] ctx CMN-600 模块上下文指针
371 * @param[in] ldid 逻辑设备 ID

373 * @param[in] local_remote 本地/远端标志（LOCAL_CCIX_NODE 或 REMOTE_CCIX_NODE）
374 */
375static void program_hnf_ldid_to_chi_node_id_reg(struct cmn600_ctx *ctx,
376 uint8_t ldid, uint32_t node_id, uint8_t local_remote)
377{
378 uint32_t reg_index = 0;
379 uint32_t reg_bit_offset = 0;
380 uint32_t i = 0;
381 struct cmn600_hnf_reg *hnf_reg = NULL;
382
383 /* 每个 64 位寄存器容纳 2 组配置数据，每组 32 位 */
384 reg_index = ldid / 2;
385 reg_bit_offset = ldid % 2;
386
387 for (i = 0; i < ctx->hnf_count; i++) {
388 hnf_reg = (struct cmn600_hnf_reg *)ctx->hnf_offset[i];
389 /* 写入 NodeID、本地/远端标志和有效位 */
390 hnf_reg->RN_PHYS_ID[reg_index] |=
391 ((uint64_t)(node_id | (local_remote <<
392 HNF_RN_PHYS_RN_LOCAL_REMOTE_SHIFT_VAL) |
393 ((uint64_t)0x1 <<
394 HNF_RN_PHYS_RN_ID_VALID_SHIFT_VAL)) <<
395 (reg_bit_offset * NUM_BITS_RESERVED_FOR_PHYS_ID));
396 }

399/**
400 * @brief 编程 CXRA SAM 地址区域寄存器
401 *
402 * @details 根据远端节点配置中的 HA 内存映射表，配置 CXRA 的 SAM 地址区域。
403 * 每个区域包含大小编码、基地址、远端 HA ID 和有效位。
404 *
405 * @param[in] ctx CMN-600 模块上下文指针
406 * @param[in] config 远端节点 CCIX 配置数据
407 *
408 * @note 区域大小必须为 64KB 的倍数且为 2 的幂次
409 * @note 区域基地址必须按区域大小自然对齐
410 */
411static void program_cxg_ra_sam_addr_region(struct cmn600_ctx *ctx,
412 struct mod_cmn600_ccix_remote_node_config *config)
413{
414 uint8_t i;
415 uint64_t blocks;
416 uint64_t sz;
417
418 for (i = 0; i < config->remote_ha_mmap_count; i++) {
419 /* 大小必须是 64KB 的倍数 */
420 fwk_assert((config->remote_ha_mmap[i].size % (64 * 1024)) == 0);
421
422 /* 大小必须是 2 的幂次 */
423 fwk_assert((config->remote_ha_mmap[i].size &
424 (config->remote_ha_mmap[i].size - 1)) == 0);
425
426 /* 区域基地址必须按区域大小自然对齐 */
427 fwk_assert(config->remote_ha_mmap[i].base %
428 config->remote_ha_mmap[i].size == 0);
429
430 blocks = config->remote_ha_mmap[i].size / (64 * 1024);
431 sz = fwk_math_log2(blocks);
432 ctx->cxg_ra_reg->CXG_RA_SAM_ADDR_REGION_REG[i] =
433 sz | (config->remote_ha_mmap[i].base) |
434 ((uint64_t)ctx->cxg_ha_id_remote << SAM_ADDR_HOME_AGENT_ID_SHIFT) |
435 (SAM_ADDR_REG_VALID_MASK);
436 }

439/**
440 * @brief 使能并启动 CCIX 链路 UP 序列
441 *
442 * @details 执行完整的 CCIX 链路建立流程：
443 * 1. 配置 CXLA 协议参数（优化 TLP、消息打包、最大包大小、PCIe 总线号、Traffic Class、Vendor ID）
444 * 2. 设置链路使能位，等待 EN 位置位
445 * 3. 验证链路 DOWN 状态（UP 位清零、DOWN 位置位、ACK 位清零）
446 * 4. 发起链路 UP 请求（REQ 位置位），等待 ACK 位置位和 DOWN 位清零
447 *
448 * @param[in] ctx CMN-600 模块上下文指针
449 * @param[in] config 远端节点 CCIX 配置数据
450 * @param[in] link_id CCIX 链路标识符（0-2）
451 *
452 * @return FWK_SUCCESS 成功；FWK_E_PARAM link_id 超出范围；
453 * 其他错误码表示链路建立超时
454 */
455static int enable_and_start_ccix_link_up_sequence(struct cmn600_ctx *ctx,
456 struct mod_cmn600_ccix_remote_node_config *config, uint8_t link_id)
457{
458 uint64_t val1;
459 int status;
460 struct cxg_wait_condition_data wait_data;
461
462 if (link_id > 2)
463 return FWK_E_PARAM;
464
465 wait_data.ctx = ctx;
466 wait_data.link_id = link_id;
467
468 /* 配置 CXLA 协议参数 */
469
470 if (config->ccix_opt_tlp)
471 ctx->cxla_reg->CXLA_CCIX_PROP_CONFIGURED |= PCIE_OPT_HDR_MASK;
472 else
473 ctx->cxla_reg->CXLA_CCIX_PROP_CONFIGURED &= ~PCIE_OPT_HDR_MASK;
474
475 if (config->ccix_msg_pack_enable) {
476 ctx->cxla_reg->CXLA_CCIX_PROP_CONFIGURED &=
477 ~(1U << CXLA_CCIX_PROP_MSG_PACK_SHIFT_VAL);
478 }
479
480 ctx->cxla_reg->CXLA_CCIX_PROP_CONFIGURED &=
481 ~CXLA_CCIX_PROP_MAX_PACK_SIZE_MASK;
482

598/**
599 * @brief CCIX 网关配置设置
600 *
601 * @details 完整的 CCIX Gateway 配置流程：
602 * 1. 如需 SMP 模式则使能
603 * 2. 为本地 RN-F/RN-D/RN-I 代理分配 RAID 并编程 LDID-to-RAID 映射
604 * 3. 为所有本地代理编程 AgentID-to-LinkID 映射
605 * 4. 配置 CXHA HA ID
606 * 5. 为远端 RN-F 代理编程 CXHA RAID-to-LDID 映射和 HN-F LDID-to-CHI-Node-ID
607 * 6. 配置 CXRA SAM 地址区域
608 * 7. 启动 CCIX 链路 UP 序列
609 */
610int ccix_setup(struct cmn600_ctx *ctx, void *remote_config)
611{
612 unsigned int i;
613 uint8_t rnf_ldid;
614 uint8_t rni_ldid;
615 uint8_t rnd_ldid;
616 uint8_t agent_id;
617 uint8_t remote_agent_id;
618 uint8_t offset_id;
619 unsigned int block;
620 uint8_t local_ra_cnt;
621 uint8_t unique_remote_rnf_ldid_value;
622 int status;
623
624 struct mod_cmn600_ccix_remote_node_config *ccix_remote_config =
625 (struct mod_cmn600_ccix_remote_node_config *)remote_config;
626
627 FWK_LOG_INFO(MOD_NAME "Programming CCIX gateway...");
628
629 /*
630 * 断言外部 RN-SAM 数量在有效范围内，
631 * 因为后续会被赋值给 uint8_t 类型
632 */
633 fwk_assert((ctx->external_rnsam_count > 1) &&
634 (ctx->external_rnsam_count <= (0xFF + 1)));
635
636 /* 本地 RA 总数 */
637 local_ra_cnt = ctx->internal_rnsam_count + ctx->external_rnsam_count;
638
639 if (ccix_remote_config->smp_mode == true) {
640 status = enable_smp_mode(ctx);
641 if (status != FWK_SUCCESS)
642 return status;
643 }
644
645 /* RAID 从 0 开始分配 */
646 ctx->raid_value = 0;
647
648 offset_id = ctx->chip_id * local_ra_cnt;
649
650 /* 为本地 RN-F 代理编程 RAID/LDID 映射 */
651 for (rnf_ldid = 0; rnf_ldid < ctx->rnf_count; rnf_ldid++) {
652 agent_id = ctx->raid_value + offset_id;
653
654 /* 编程 CXRA LDID-to-RAID 查找表 */
655 program_cxg_ra_rnf_ldid_to_raid_reg(ctx, rnf_ldid, agent_id);
656
657 /*
658 * 编程 CXRA/CXHA/CXLA 中的 AgentID-to-LinkID 查找表，
659 * 用于远端代理的路由
660 */
661 program_agentid_to_linkid_reg(ctx, agent_id,
662 ccix_remote_config->ccix_link_id);
663
664 /*
665 * 本地 RN-F 代理的 HN-F LDID-to-CHI-Node-ID 有效位
666 * 已在初始化阶段编程
667 */
668 program_cxg_ha_id(ctx, ccix_remote_config->remote_ha_mmap[0].ha_id);
669 }
670
671 /*
672 * unique_remote_rnf_ldid_value 用于跟踪远端 RN-F 代理的 LDID
673 */
674 unique_remote_rnf_ldid_value = ctx->rnf_count;
675
676 /* 为远端 RN-F 代理编程 RAID/LDID 映射 */
677 for (i = 0; i < ccix_remote_config->remote_rnf_count; i++) {
678 block = i / ctx->rnf_count;
679
680 /*
681 * 远端 agent_id 不应包含当前芯片的代理 ID。
682 * 若 block 小于当前 chip_id，则包含芯片 0 到当前芯片（不含）的代理 ID；
683 * 若 block 大于等于当前 chip_id，则包含下一个芯片到最大芯片的代理 ID。
684 */
685 if (block < ctx->chip_id)
686 remote_agent_id = i + (ctx->rnf_count * block);
687 else
688 remote_agent_id = i + (ctx->rnf_count * block) + local_ra_cnt;
689
690 /* 编程 CXHA RAID-to-LDID 查找表 */
691 program_cxg_ha_raid_to_ldid_lut(ctx, remote_agent_id,
692 unique_remote_rnf_ldid_value);
693 /*
694 * 编程 HN-F LDID-to-CHI-Node-ID，
695 * 用于远端 RN-F 代理
696 */
697 program_hnf_ldid_to_chi_node_id_reg(ctx,
698 unique_remote_rnf_ldid_value, ctx->cxg_ha_node_id,
699 REMOTE_CCIX_NODE);
700
701 unique_remote_rnf_ldid_value++;
702 }
703
704 offset_id = ctx->chip_id * local_ra_cnt;
705
706 /* 为本地 RN-D 代理编程 RAID/LDID 映射 */
707 for (i = 0; i < ctx->rnd_count; i++) {
708 rnd_ldid = ctx->rnd_ldid[i];
709
710 /* 计算远端代理的 agent_id */
711 agent_id = ctx->raid_value + offset_id;
712
713 /* 编程 CXRA LDID-to-RAID 查找表 */
714 program_cxg_ra_rnd_ldid_to_raid_reg(ctx, rnd_ldid, agent_id);
715
716 /* 编程远端代理的 AgentID-to-LinkID 查找表 */
717 program_agentid_to_linkid_reg(ctx, agent_id,
718 ccix_remote_config->ccix_link_id);
719 }
720
721 /* 为本地 RN-I 代理编程 RAID/LDID 映射 */
722 for (i = 0; i < ctx->rni_count; i++) {
723 rni_ldid = ctx->rni_ldid[i];
724
725 /* 计算远端代理的 agent_id */
726 agent_id = ctx->raid_value + offset_id;
727
728 /* 编程 CXRA LDID-to-RAID 查找表 */
729 program_cxg_ra_rni_ldid_to_raid_reg(ctx, rni_ldid, agent_id);
730
731 /* 编程远端代理的 AgentID-to-LinkID 查找表 */
732 program_agentid_to_linkid_reg(ctx, agent_id,
733 ccix_remote_config->ccix_link_id);
734 }
735
736 program_cxg_ra_sam_addr_region(ctx, ccix_remote_config);
737 status = enable_and_start_ccix_link_up_sequence(ctx, ccix_remote_config,
738 ccix_remote_config->ccix_link_id);
739
740 return status;
741}
