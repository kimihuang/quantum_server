27/**

29 *
30 * @details 用于在运行时记录需要配置 RN-SAM 映射的外部节点。

39};

57/**
58 * @brief CMN-600 驱动模块运行时上下文
59 *
60 * @details 保存 CMN-600 互连拓扑发现结果、各类节点信息、
61 * CCIX 配置状态及模块依赖的 API。该结构体在模块初始化时
62 * 通过 fwk_mm_calloc 分配，整个模块生命周期内唯一。
63 */
64struct cmn600_ctx {
65 /** @brief 指向模块配置数据的指针，由 fwk 在 init 时传入 */
66 const struct mod_cmn600_config *config;
67
68 /** @brief 指向根节点（Configuration Manager）寄存器的指针 */
69 struct cmn600_cfgm_reg *root;
70
71 /** @brief 系统中 HN-F（系统缓存）节点的数量 */
72 unsigned int hnf_count;
73
74 /** @brief HN-F 节点寄存器偏移量数组，每个元素为对应 HN-F 的寄存器基地址偏移 */
75 uint32_t hnf_offset[MAX_HNF_COUNT];
76
77 /** @brief HN-F 缓存分组描述符数组，用于配置 RN-SAM 的 SYS_CACHE_GRP_HN_NODEID 寄存器 */
78 uint64_t *hnf_cache_group;
79
80 /**
81 * @brief 外部 RN-SAM 节点数量
82 *
83 * @details 外部 RN-SAM 节点位于当前 CMN-600 实例的时钟/电源域之外，
84 * 需要在对应电源域上电后通过 PPU 观察者回调进行配置。
85 */
86 unsigned int external_rnsam_count;
87
88 /** @brief 外部 RN-SAM 节点元组表，包含节点 ID 和寄存器指针 */
89 struct external_rnsam_tuple *external_rnsam_table;
90
91 /**
92 * @brief 内部 RN-SAM 节点数量
93 *
94 * @details 内部 RN-SAM 节点位于当前 CMN-600 实例内，
95 * 在系统拓扑发现完成后统一配置。
96 */
97 unsigned int internal_rnsam_count;
98
99 /** @brief 内部 RN-SAM 节点寄存器指针数组 */
100 struct cmn600_rnsam_reg **internal_rnsam_table;
101
102 /**
103 * @brief RN-D 节点数量
104 *
105 * @details 驱动保存 RN-D 节点的 LDID 列表，
106 * 用于 CCIX 配置时建立 LDID 到 RAID 的映射。
107 */
108 unsigned int rnd_count;
109
110 /** @brief RN-D 节点的 LDID 数组 */
111 uint8_t rnd_ldid[MAX_RND_COUNT];
112
113 /**
114 * @brief RN-F 节点数量
115 *
116 * @details RN-F 节点没有独立的节点类型标识符，
117 * 其数量通过 RN-SAM 总数减去 RN-D、RN-I 和 CXHA 数量推算。
118 */
119 unsigned int rnf_count;
120
121 /**
122 * @brief RN-I 节点数量
123 *
124 * @details 驱动保存 RN-I 节点的 LDID 列表，
125 * 用于 CCIX 配置时建立 LDID 到 RAID 的映射。
126 */
127 unsigned int rni_count;
128
129 /** @brief RN-I 节点的 LDID 数组 */
130 uint8_t rni_ldid[MAX_RNI_COUNT];
131
132 /** @brief CCIX Home Agent 的本地 HA ID */
133 unsigned int cxg_ha_id;
134
135 /** @brief CCIX Home Agent 的节点 ID */
136 unsigned int cxg_ha_node_id;
137
138 /** @brief CCIX 远端 Home Agent 的 HA ID */
139 unsigned int cxg_ha_id_remote;
140
141 /** @brief 当前 RAID (Requesting Agent ID) 分配值，用于 CCIX 配置时递增分配 */
142 uint8_t raid_value;
143
144 /** @brief 指向 CCIX Gateway Requesting Agent 寄存器的指针 */
145 struct cmn600_cxg_ra_reg *cxg_ra_reg;
146
147 /** @brief 指向 CCIX Gateway Home Agent 寄存器的指针 */
148 struct cmn600_cxg_ha_reg *cxg_ha_reg;
149
150 /** @brief 指向 CCIX Gateway Link Agent 寄存器的指针 */
151 struct cmn600_cxla_reg *cxla_reg;
152
153 /** @brief CCIX 主节点配置信息，用于传递给上层固件（如 SCMI） */
154 struct mod_cmn600_ccix_host_node_config ccix_host_info;
155
156 /** @brief 定时器模块 API 指针，用于 CCIX 链路建立过程中的超时等待 */
157 struct mod_timer_api *timer_api;
158
159 /** @brief 系统信息指针，包含多芯片模式和 chip ID 等信息 */
160 const struct mod_system_info *system_info;
161
162 /** @brief 模块初始化完成标志，防止重复执行发现和配置流程 */
163 bool initialized;
164
165 /** @brief 当前芯片的 chip ID，用于多芯片场景下的地址偏移计算 */
166 uint8_t chip_id;
167};
168
169/**
170 * @brief 配置 RN-SAM（请求节点系统地址映射）
171 *
172 * @details 根据模块配置中的内存区域映射表，对指定的 RN-SAM 节点
173 * 进行地址映射配置，包括 IO 区域、系统缓存区域和 CCIX 区域。
174 *
175 * @param[in] rnsam 指向待配置的 RN-SAM 寄存器结构体
176 *
177 * @return 0 成功（FWK_SUCCESS）；负值为错误码
178 * @retval FWK_E_DATA 内存区域数量超过硬件限制
179 * @retval FWK_E_DEVICE 不支持的 CMN-600 修订版本
180 *
181 * @pre ctx 必须已初始化，且 cmn600_discovery() 已成功执行
182 * @post RN-SAM 寄存器已配置完成并使能（STATUS 寄存器写入 UNSTALL）
183 */
184int cmn600_setup_sam(struct cmn600_rnsam_reg *rnsam);
185
186#endif /* INTERNAL_CMN600_CTX_H */
