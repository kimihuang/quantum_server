83/**
84 * @brief 处理 HN-F（全一致性主节点/系统缓存）节点
85 *
86 * @details 配置 HN-F 的缓存分组、SAM 目标节点、子区域映射和 PPU 电源策略。
87 * 在 CAL 模式下，仅将偶数编号的 HN-F 加入缓存分组。
88 *
89 * @param[in] hnf 指向 HN-F 寄存器结构体的指针
90 *
91 * @note CAL 模式下 HN-F 数量必须为偶数
92 */
93static void process_node_hnf(struct cmn600_hnf_reg *hnf)
94{
95 unsigned int logical_id;
96 unsigned int node_id;
97 unsigned int group;
98 unsigned int bit_pos;
99 unsigned int region_idx;
100 unsigned int region_sub_count = 0;
101 const struct mod_cmn600_memory_region_map *region;
102 const struct mod_cmn600_config *config = ctx->config;
103 static unsigned int cal_mode_factor = 1;
104 uint64_t base_offset;
105
106 logical_id = get_node_logical_id(hnf);
107 node_id = get_node_id(hnf);
108
109 /*
110 * CAL 模式下，只有偶数编号的 HN-F 节点应被添加到
111 * sys_cache_grp_hn_nodeid 寄存器，且 hnf_count 仅对偶数节点递增。
112 */
113 if (config->hnf_cal_mode == true && (node_id % 2 == 1) &&
114 is_cal_mode_supported(ctx->root)) {
115
116 /* 调整分组和位偏移计算的因子 */
117 cal_mode_factor = 2;
118
119 /*
120 * 减少计数，因为当前 HN-F 节点不会被包含在
121 * sys_cache_grp_hn_nodeid 寄存器中
122 */
123 ctx->hnf_count--;
124 }
125
126 fwk_assert(logical_id < config->snf_count);
127
128 group = logical_id /
129 (CMN600_HNF_CACHE_GROUP_ENTRIES_PER_GROUP * cal_mode_factor);
130 bit_pos = (CMN600_HNF_CACHE_GROUP_ENTRY_BITS_WIDTH / cal_mode_factor) *
131 (logical_id % (CMN600_HNF_CACHE_GROUP_ENTRIES_PER_GROUP *
132 cal_mode_factor));
133
134 /*
135 * CAL 模式下，仅将偶数编号的 HN-F 节点添加到
136 * sys_cache_grp_hn_nodeid 寄存器
137 */
138 if (config->hnf_cal_mode == true && is_cal_mode_supported(ctx->root)) {
139 if (node_id % 2 == 0)
140 ctx->hnf_cache_group[group] += ((uint64_t)get_node_id(hnf)) <<
141 bit_pos;
142 } else
143 ctx->hnf_cache_group[group] += ((uint64_t)get_node_id(hnf)) << bit_pos;
144
145 /* 设置目标 SN-F 节点 */
146 hnf->SAM_CONTROL = config->snf_table[logical_id];
147
148 if (ctx->chip_id != 0) {
149 base_offset = ((uint64_t)(ctx->config->chip_addr_space *
150 ctx->chip_id));
151 } else
152 base_offset = 0;
153
154 /*
155 * 将子区域映射到此 HN-F 节点
156 */
157 for (region_idx = 0; region_idx < config->mmap_count; region_idx++) {
158 region = &config->mmap_table[region_idx];
159
160 /* 跳过非子区域 */
161 if (region->type != MOD_CMN600_REGION_TYPE_SYSCACHE_SUB)
162 continue;
163
164 /* 配置子区域条目 */
165 hnf->SAM_MEMREGION[region_sub_count] = region->node_id |
166 (sam_encode_region_size(region->size) <<
167 CMN600_HNF_SAM_MEMREGION_SIZE_POS) |
168 (((region->base + base_offset) / SAM_GRANULARITY) <<
169 CMN600_HNF_SAM_MEMREGION_BASE_POS) |
170 CMN600_HNF_SAM_MEMREGION_VALID;
171
172 region_sub_count++;
173 }
174
175 /* 配置系统缓存 RAM PPU：开启电源，全关联模式，动态电源门控使能 */
176 hnf->PPU_PWPR = CMN600_PPU_PWPR_POLICY_ON |
177 CMN600_PPU_PWPR_OPMODE_FAM |
178 CMN600_PPU_PWPR_DYN_EN;
179}

181/**
182 * @brief 扫描 CMN-600 互连拓扑
183 *
184 * @details 遍历 mesh 中所有 XP 和其子节点，统计：
185 * - 外部 RN-SAM 节点数量
186 * - 内部 RN-SAM 节点数量
187 * - HN-F 节点数量
188 * - RN-D、RN-I 节点数量
189 * - CCIX 相关节点（CXRA、CXHA、CXLA）
190 * RN-F 数量通过 RN-SAM 总数减去 RN-D、RN-I 和 CXHA 数量推算。
191 *
192 * @return FWK_SUCCESS 成功；FWK_E_DATA 节点数量超限；FWK_E_RANGE RN-F 数量超限
193 */
194static int cmn600_discovery(void)
195{
196 unsigned int xp_count;
197 unsigned int xp_idx;
198 unsigned int node_count;
199 unsigned int node_idx;
200 bool xp_port;
201 struct cmn600_mxp_reg *xp;
202 struct node_header *node;
203 const struct mod_cmn600_config *config = ctx->config;
204
205 FWK_LOG_INFO(
206 MOD_NAME "CMN-600 revision: %s", get_cmn600_revision_name(ctx->root));
207 FWK_LOG_INFO(MOD_NAME "Starting discovery...");
208
209 fwk_assert(get_node_type(ctx->root) == NODE_TYPE_CFG);
210
211 /* 遍历交叉点 (XP) */
212 xp_count = get_node_child_count(ctx->root);
213 for (xp_idx = 0; xp_idx < xp_count; xp_idx++) {
214
215 xp = get_child_node(config->base, ctx->root, xp_idx);
216 fwk_assert(get_node_type(xp) == NODE_TYPE_XP);
217
218 FWK_LOG_INFO(MOD_NAME);
219
220 FWK_LOG_INFO(
221 MOD_NAME "XP (%d, %d) ID:%d, LID:%d",
222 get_node_pos_x(xp),
223 get_node_pos_y(xp),
224 get_node_id(xp),
225 get_node_logical_id(xp));
226
227 /* 遍历 XP 下的节点 */
228 node_count = get_node_child_count(xp);
229 for (node_idx = 0; node_idx < node_count; node_idx++) {
230
231 node = get_child_node(config->base, xp, node_idx);
232
233 /* 外部节点 */
234 if (is_child_external(xp, node_idx)) {
235 xp_port = get_port_number(get_child_node_id(xp, node_idx));
236
237 /*
238 * 如果设备类型为 CXRH、CXHA 或 CXRA，则外部子节点
239 * 为 CXLA，因为每个 CXRH/CXHA/CXRA 节点都有一个
240 * 对应的外部 CXLA 节点。
241 */
242 if ((get_device_type(xp, xp_port) == DEVICE_TYPE_CXRH) ||
243 (get_device_type(xp, xp_port) == DEVICE_TYPE_CXHA) ||
244 (get_device_type(xp, xp_port) == DEVICE_TYPE_CXRA)) {
245 ctx->cxla_reg = (void *)node;
246 FWK_LOG_INFO(
247 MOD_NAME " Found CXLA at node ID: %d",
248 get_child_node_id(xp, node_idx));
249 } else { /* 外部 RN-SAM 节点 */
250 ctx->external_rnsam_count++;
251 FWK_LOG_INFO(
252 MOD_NAME " Found external node ID: %d",
253 get_child_node_id(xp, node_idx));
254 }
255 } else { /* 内部节点 */
256 switch (get_node_type(node)) {
257 case NODE_TYPE_HN_F:
258 if (ctx->hnf_count >= MAX_HNF_COUNT) {
259 FWK_LOG_INFO(
260 MOD_NAME " hnf count %d >= max limit (%d)",
261 ctx->hnf_count,
262 MAX_HNF_COUNT);
263 return FWK_E_DATA;
264 }
265 ctx->hnf_offset[ctx->hnf_count++] = (uint32_t)node;
266 break;
267
268 case NODE_TYPE_RN_SAM:
269 ctx->internal_rnsam_count++;
270 break;
271
272 case NODE_TYPE_RN_D:
273 if (ctx->rnd_count >= MAX_RND_COUNT) {
274 FWK_LOG_INFO(
275 MOD_NAME " rnd count %d >= max limit (%d)",
276 ctx->rnd_count,
277 MAX_RND_COUNT);
278 return FWK_E_DATA;
279 }
280 ctx->rnd_ldid[ctx->rnd_count++] = get_node_logical_id(node);
281 break;
282
283 case NODE_TYPE_RN_I:
284 if (ctx->rni_count >= MAX_RNI_COUNT) {
285 FWK_LOG_INFO(
286 MOD_NAME " rni count %d >= max limit (%d)",
287 ctx->rni_count,
288 MAX_RNI_COUNT);
289 return FWK_E_DATA;
290 }
291 ctx->rni_ldid[ctx->rni_count++] = get_node_logical_id(node);
292 break;
293
294 case NODE_TYPE_CXRA:
295 ctx->cxg_ra_reg = (struct cmn600_cxg_ra_reg *)node;
296 break;
297
298 case NODE_TYPE_CXHA:
299 ctx->cxg_ha_reg = (struct cmn600_cxg_ha_reg *)node;
300 ctx->ccix_host_info.host_ha_count++;
301 break;
302
303 default:
304 /* 其他节点类型无需处理 */
305 break;
306 }
307
308 FWK_LOG_INFO(
309 MOD_NAME " %s ID:%d, LID:%d",
310 get_node_type_name(get_node_type(node)),
311 get_node_id(node),
312 get_node_logical_id(node));
313 }
314 }
315 }
316
317 /*
318 * RN-F 节点没有节点类型标识符，因此无法在发现过程中直接计数。
319 * RN-F 数量 = RN-SAM 总数 - RN-D 总数 - RN-I 总数 - CXHA 总数
320 */
321 ctx->rnf_count = ctx->internal_rnsam_count + ctx->external_rnsam_count -
322 (ctx->rnd_count + ctx->rni_count + ctx->ccix_host_info.host_ha_count);
323
324 if (ctx->rnf_count > MAX_RNF_COUNT) {
325 FWK_LOG_ERR(
326 MOD_NAME "rnf count %d > max limit (%d)",
327 ctx->rnf_count,
328 MAX_RNF_COUNT);
329 return FWK_E_RANGE;
330 }
331
332 /* CAL 模式下 HN-F 数量必须为偶数 */
333 if ((ctx->hnf_count % 2 != 0) && (config->hnf_cal_mode == true)) {
334 FWK_LOG_ERR(
335 MOD_NAME "hnf count: %d should be even when CAL mode is set",
336 ctx->hnf_count);
337 return FWK_E_DATA;
338 }
339
340 FWK_LOG_INFO(
341 MOD_NAME "Total internal RN-SAM nodes: %d", ctx->internal_rnsam_count);
342 FWK_LOG_INFO(
343 MOD_NAME "Total external RN-SAM nodes: %d", ctx->external_rnsam_count);
344 FWK_LOG_INFO(MOD_NAME "Total HN-F nodes: %d", ctx->hnf_count);
345 FWK_LOG_INFO(MOD_NAME "Total RN-F nodes: %d", ctx->rnf_count);
346 FWK_LOG_INFO(MOD_NAME "Total RN-D nodes: %d", ctx->rnd_count);
347 FWK_LOG_INFO(MOD_NAME "Total RN-I nodes: %d", ctx->rni_count);
348
349 if (ctx->cxla_reg != NULL) {
350 FWK_LOG_INFO(MOD_NAME "CCIX CXLA node at: 0x%p", (void *)ctx->cxla_reg);
351 }
352 if (ctx->cxg_ra_reg != NULL) {
353 FWK_LOG_INFO(
354 MOD_NAME "CCIX CXRA node at: 0x%p", (void *)ctx->cxg_ra_reg);
355 }
356 if (ctx->cxg_ha_reg != NULL) {
357 FWK_LOG_INFO(
358 MOD_NAME "CCIX CXHA node at: 0x%p", (void *)ctx->cxg_ha_reg);
359 }
360 return FWK_SUCCESS;
361}

363/**
364 * @brief 配置 CMN-600 互连中的所有节点
365 *
366 * @details 第二遍遍历 mesh，填充外部/内部 RN-SAM 表，
367 * 并对每个 HN-F 节点执行配置（process_node_hnf）。
368 * 此函数在 cmn600_discovery() 之后调用，
369 * 此时资源已分配完毕。
370 */
371static void cmn600_configure(void)
372{
373 unsigned int xp_count;
374 unsigned int xp_idx;
375 bool xp_port;
376 unsigned int node_count;
377 unsigned int node_idx;
378 unsigned int xrnsam_entry;
379 unsigned int irnsam_entry;
380 struct cmn600_mxp_reg *xp;
381 void *node;
382 const struct mod_cmn600_config *config = ctx->config;
383
384 fwk_assert(get_node_type(ctx->root) == NODE_TYPE_CFG);
385
386 xrnsam_entry = 0;
387 irnsam_entry = 0;
388
389 /* 遍历交叉点 (XP) */
390 xp_count = get_node_child_count(ctx->root);
391 for (xp_idx = 0; xp_idx < xp_count; xp_idx++) {
392 xp = get_child_node(config->base, ctx->root, xp_idx);
393 fwk_assert(get_node_type(xp) == NODE_TYPE_XP);
394
395 /* 遍历节点 */
396 node_count = get_node_child_count(xp);
397 for (node_idx = 0; node_idx < node_count; node_idx++) {
398 node = get_child_node(config->base, xp, node_idx);
399
400 if (is_child_external(xp, node_idx)) {
401 unsigned int node_id = get_child_node_id(xp, node_idx);
402 xp_port = get_port_number(get_child_node_id(xp, node_idx));
403
404 /* 跳过 CCIX 设备类型 */
405 if ((get_device_type(xp, xp_port) == DEVICE_TYPE_CXRH) ||
406 (get_device_type(xp, xp_port) == DEVICE_TYPE_CXHA) ||
407 (get_device_type(xp, xp_port) == DEVICE_TYPE_CXRA))
408 continue;
409
410 fwk_assert(xrnsam_entry < ctx->external_rnsam_count);
411
412 ctx->external_rnsam_table[xrnsam_entry].node_id = node_id;
413 ctx->external_rnsam_table[xrnsam_entry].node = node;
414
415 xrnsam_entry++;
416 } else {
417 enum node_type node_type = get_node_type(node);
418
419 if (node_type == NODE_TYPE_RN_SAM) {
420 fwk_assert(irnsam_entry < ctx->internal_rnsam_count);
421
422 ctx->internal_rnsam_table[irnsam_entry] = node;
423
424 irnsam_entry++;
425 } else if (node_type == NODE_TYPE_HN_F)
426 process_node_hnf(node);
427 }
428 }
429 }
430}

432/**
433 * @brief 获取非哈希目标节点 ID 寄存器数量
434 *
435 * @details 不同修订版本的 CMN-600 拥有不同数量的 NON_HASH_TGT_NODEID 寄存器：
436 * - r1p0/r1p1/r1p3: 2 个
437 * - r1p2/r2p0: 3 个
438 * - r3p0 及以上: 5 个
439 *
440 * @return 寄存器数量（正值）；FWK_E_DEVICE 不支持的修订版本
441 */
442static int get_tgt_nodeid_reg_count()
443{
444 switch (get_cmn600_revision(ctx->root)) {
445 case CMN600_PERIPH_ID_2_REV_R1_P0:
446 case CMN600_PERIPH_ID_2_REV_R1_P1:
447 case CMN600_PERIPH_ID_2_REV_R1_P3:
448 return CMN600_RNSAM_NON_HASH_TGT_NODEID_REGS_2;
449
450 case CMN600_PERIPH_ID_2_REV_R1_P2:
451 case CMN600_PERIPH_ID_2_REV_R2_P0:
452 return CMN600_RNSAM_NON_HASH_TGT_NODEID_REGS_3;
453
454 case CMN600_PERIPH_ID_2_REV_R3_P0:
455 case CMN600_PERIPH_ID_2_REV_R3_P1:
456 case CMN600_PERIPH_ID_2_REV_R3_P2:
457 return CMN600_RNSAM_NON_HASH_TGT_NODEID_REGS_5;
458
459 default:
460 fwk_unexpected();
461 return FWK_E_DEVICE;
462 }
463}

465/**
466 * @brief 配置 RN-SAM（请求节点系统地址映射）
467 *
468 * @details 根据模块配置中的内存区域映射表，配置指定 RN-SAM 节点的地址映射，
469 * 包括 IO 区域、CCIX 区域、系统缓存区域和非哈希系统缓存区域。
470 * 配置完成后使能 RN-SAM。
471 */
472int cmn600_setup_sam(struct cmn600_rnsam_reg *rnsam)
473{
474 unsigned int region_idx;
475 unsigned int region_io_count = 0;
476 unsigned int region_sys_count = 0;
477 const struct mod_cmn600_memory_region_map *region;
478 const struct mod_cmn600_config *config = ctx->config;
479 unsigned int bit_pos;
480 unsigned int group;
481 unsigned int group_count;
482 enum sam_node_type sam_node_type;
483 uint64_t base;
484 unsigned int scg_region = 0;
485 unsigned int scg_regions_enabled[CMN600_MAX_NUM_SCG] = {0, 0, 0, 0};
486 unsigned int tgt_nodeid_reg_count;
487 volatile uint64_t *nodeid;
488 int status;
489
490 FWK_LOG_INFO(MOD_NAME "Configuring SAM for node %d", get_node_id(rnsam));
491
492 status = get_tgt_nodeid_reg_count();
493 if (status < 0)
494 return FWK_E_DEVICE;
495 tgt_nodeid_reg_count = (unsigned int)status;
496
497 for (region_idx = 0; region_idx < config->mmap_count; region_idx++) {
498 region = &config->mmap_table[region_idx];
499
500 /* 多芯片场景下的基地址偏移处理 */
501 if (ctx->chip_id != 0) {
502 if (region->type == MOD_CMN600_REGION_TYPE_CCIX)
503 base = 0;
504 else if (region->type == MOD_CMN600_MEMORY_REGION_TYPE_SYSCACHE)
505 base = region->base;
506 else {
507 base = ((uint64_t)(ctx->config->chip_addr_space *
508 ctx->chip_id) + region->base);
509 }
510 } else
511 base = region->base;
512
513 (void)mmap_type_name;
514 FWK_LOG_INFO(
515 MOD_NAME " [0x%08" PRIX32 "%08" PRIX32
516 " - 0x%08" PRIX32 "%08" PRIX32 "] %s",
517 (uint32_t)(base >> 32),
518 (uint32_t)base,
519 (uint32_t)((base + region->size - 1) >> 32),
520 (uint32_t)(base + region->size - 1),
521 mmap_type_name[region->type]);
522
523 switch (region->type) {
524 case MOD_CMN600_MEMORY_REGION_TYPE_IO:
525 case MOD_CMN600_REGION_TYPE_CCIX:
526 /*
527 * 配置非哈希内存区域（IO 或 CCIX）
528 */
529 if (region_io_count > CMN600_RNSAM_MAX_NON_HASH_MEM_REGION_ENTRIES(
530 tgt_nodeid_reg_count)) {
531 FWK_LOG_ERR(
532 MOD_NAME
533 "Non-Hashed Memory can have maximum of %d regions only",
534 CMN600_RNSAM_MAX_NON_HASH_MEM_REGION_ENTRIES(
535 tgt_nodeid_reg_count));
536 return FWK_E_DATA;
537 }
538
539 group = region_io_count / CMN600_RNSAM_REGION_ENTRIES_PER_GROUP;
540 bit_pos = (region_io_count %
541 CMN600_RNSAM_REGION_ENTRIES_PER_GROUP) *
542 CMN600_RNSAM_REGION_ENTRY_BITS_WIDTH;
543
544 sam_node_type =
545 (region->type == MOD_CMN600_MEMORY_REGION_TYPE_IO) ?
546 SAM_NODE_TYPE_HN_I : SAM_NODE_TYPE_CXRA;
547 configure_region(&rnsam->NON_HASH_MEM_REGION[group],
548 bit_pos,
549 base,
550 region->size,
551 sam_node_type);
552
553 /*
554 * 配置目标节点 ID
555 */
556 group = region_io_count /
557 CMN600_RNSAM_NON_HASH_TGT_NODEID_ENTRIES_PER_GROUP;
558 if (group > tgt_nodeid_reg_count) {
559 FWK_LOG_ERR(
560 MOD_NAME
561 "Non-Hashed Memory can have maximum of %d groups only",
562 tgt_nodeid_reg_count);
563 return FWK_E_DATA;
564 }
565
566 bit_pos = CMN600_RNSAM_NON_HASH_TGT_NODEID_ENTRY_BITS_WIDTH *
567 (region_io_count %
568 CMN600_RNSAM_NON_HASH_TGT_NODEID_ENTRIES_PER_GROUP);
569
570 if (group < CMN600_RNSAM_NON_HASH_TGT_NODEID_REGS_3) {
571 nodeid = &rnsam->NON_HASH_TGT_NODEID[group];
572 } else if (group < CMN600_RNSAM_NON_HASH_TGT_NODEID_REGS_5) {
573 /*
574 * CMN-600 R3P0 及以上版本有 5 个 NON_HASH_TGT_NODEID 寄存器，
575 * 最后两个寄存器位于不同的偏移地址
576 */
577 nodeid = &rnsam->NON_HASH_TGT_NODEID_GRP2
578 [group - CMN600_RNSAM_NON_HASH_TGT_NODEID_REGS_3];
579 } else {
580 FWK_LOG_ERR(MOD_NAME "Invalid Non-Hashed target node ID group");
581 return FWK_E_DATA;
582 }
583
584 *nodeid &=
585 ~(CMN600_RNSAM_NON_HASH_TGT_NODEID_ENTRY_MASK << bit_pos);
586 *nodeid |=
587 (region->node_id & CMN600_RNSAM_NON_HASH_TGT_NODEID_ENTRY_MASK)
588 << bit_pos;
589
590 region_io_count++;
591 break;
592
593 case MOD_CMN600_MEMORY_REGION_TYPE_SYSCACHE:
594 /*
595 * 配置系统缓存（哈希）内存区域
596 */
597 if (region_sys_count >= CMN600_RNSAM_MAX_HASH_MEM_REGION_ENTRIES) {
598 FWK_LOG_ERR(
599 MOD_NAME
600 "Hashed Memory can have maximum of %d regions only",
601 CMN600_RNSAM_MAX_HASH_MEM_REGION_ENTRIES);
602 return FWK_E_DATA;
603 }
604
605 group = region_sys_count / CMN600_RNSAM_REGION_ENTRIES_PER_GROUP;
606 bit_pos = (region_sys_count %
607 CMN600_RNSAM_REGION_ENTRIES_PER_GROUP) *
608 CMN600_RNSAM_REGION_ENTRY_BITS_WIDTH;
609 configure_region(&rnsam->SYS_CACHE_GRP_REGION[group],
610 bit_pos,
611 region->base,
612 region->size,
613 SAM_NODE_TYPE_HN_F);
614
615 /* 标记对应的系统缓存分组区域为已使能 */
616 scg_region = (2 * group) + (bit_pos/32);
617 fwk_assert(scg_region < CMN600_MAX_NUM_SCG);
618 scg_regions_enabled[scg_region] = 1;
619
620 region_sys_count++;
621 break;
622
623 case MOD_CMN600_REGION_TYPE_SYSCACHE_NONHASH:
624 group = region_sys_count / CMN600_RNSAM_REGION_ENTRIES_PER_GROUP;
625 bit_pos = (region_sys_count %
626 CMN600_RNSAM_REGION_ENTRIES_PER_GROUP) *
627 CMN600_RNSAM_REGION_ENTRY_BITS_WIDTH;
628 /*
629 * 配置非哈希系统缓存区域
630 */
631 configure_region(&rnsam->SYS_CACHE_GRP_REGION[group],
632 bit_pos,
633 region->base,
634 region->size,
635 SAM_NODE_TYPE_HN_I);
636
637 rnsam->SYS_CACHE_GRP_REGION[group] |= (UINT64_C(0x2) << bit_pos);
638 bit_pos = CMN600_RNSAM_NON_HASH_TGT_NODEID_ENTRY_BITS_WIDTH *
639 ((region_sys_count - 1) %
640 CMN600_RNSAM_NON_HASH_TGT_NODEID_ENTRIES_PER_GROUP);
641 rnsam->SYS_CACHE_GRP_NOHASH_NODEID &=
642 ~(CMN600_RNSAM_NON_HASH_TGT_NODEID_ENTRY_MASK << bit_pos);
643 rnsam->SYS_CACHE_GRP_NOHASH_NODEID |= (region->node_id &
644 CMN600_RNSAM_NON_HASH_TGT_NODEID_ENTRY_MASK) << bit_pos;
645
646 region_sys_count++;
647 break;
648
649 case MOD_CMN600_REGION_TYPE_SYSCACHE_SUB:
650 /* 子区域由 HN-F 节点处理，此处不做操作 */
651 break;
652
653 default:
654 fwk_unexpected();
655 return FWK_E_DATA;
656 }
657 }
658
659 /* 写入 HN-F 缓存分组节点 ID */
660 group_count = cmn600_hnf_cache_group_count(ctx->hnf_count);
661 for (group = 0; group < group_count; group++)
662 rnsam->SYS_CACHE_GRP_HN_NODEID[group] = ctx->hnf_cache_group[group];
663
664 /* 写入 HN-F 节点总数 */
665 rnsam->SYS_CACHE_GRP_HN_COUNT = ctx->hnf_count;
666
667 /* 仅 CMN-600 r2p0 及以上版本支持 CAL 模式 */
668 if (is_cal_mode_supported(ctx->root) && config->hnf_cal_mode) {
669 for (region_idx = 0; region_idx < CMN600_MAX_NUM_SCG; region_idx++)
670 rnsam->SYS_CACHE_GRP_CAL_MODE |= scg_regions_enabled[region_idx] *
671 (CMN600_RNSAM_SCG_HNF_CAL_MODE_EN <<
672 (region_idx * CMN600_RNSAM_SCG_HNF_CAL_MODE_SHIFT));
673 }
674
675 /* 使能 RN-SAM */
676 rnsam->STATUS = CMN600_RNSAM_STATUS_UNSTALL;
677 __sync_synchronize();
678
679 return FWK_SUCCESS;
680}

682/**
683 * @brief 执行 CMN-600 完整初始化流程
684 *
685 * @details 执行拓扑发现、资源分配、节点配置和 SAM 配置。
686 * 如果已初始化则跳过发现和资源分配步骤。
687 * 同时收集 CCIX 主节点拓扑信息。
688 *
689 * @return FWK_SUCCESS 成功；其他错误码表示失败
690 */
691static int cmn600_setup(void)
692{
693 unsigned int rnsam_idx, i, ccix_mmap_idx;
694 int status;
695
696 if (!ctx->initialized) {
697 status = cmn600_discovery();
698 if (status != FWK_SUCCESS)
699 return status;
700 /*
701 * 根据发现结果分配资源
702 */
703
704 /* 内部 RN-SAM 节点指针数组 */
705 if (ctx->internal_rnsam_count != 0) {
706 ctx->internal_rnsam_table = fwk_mm_calloc(
707 ctx->internal_rnsam_count, sizeof(*ctx->internal_rnsam_table));
708 }
709
710 /* 外部 RN-SAM 节点元组数组（含节点 ID） */
711 if (ctx->external_rnsam_count != 0) {
712 ctx->external_rnsam_table = fwk_mm_calloc(
713 ctx->external_rnsam_count, sizeof(*ctx->external_rnsam_table));
714 }
715
716 /* 缓存分组描述符数组 */
717 if (ctx->hnf_count != 0) {
718 /*
719 * 分配足够的分组描述符以容纳系统中所有 HN-F 节点
720 */
721 ctx->hnf_cache_group = fwk_mm_calloc(
722 cmn600_hnf_cache_group_count(ctx->hnf_count),
723 sizeof(*ctx->hnf_cache_group));
724 }
725 }
726
727 cmn600_configure();
728
729 /* 配置内部 RN-SAM 节点 */
730 for (rnsam_idx = 0; rnsam_idx < ctx->internal_rnsam_count; rnsam_idx++)
731 cmn600_setup_sam(ctx->internal_rnsam_table[rnsam_idx]);
732
733 /* 收集 CCIX 主节点拓扑信息 */
734 for (i = 0; i < ctx->config->mmap_count; i++) {
735 if (ctx->config->mmap_table[i].type == MOD_CMN600_REGION_TYPE_CCIX) {
736 ccix_mmap_idx = ctx->ccix_host_info.ccix_host_mmap_count;
737 if (ccix_mmap_idx >= MAX_HA_MMAP_ENTRIES)
738 return FWK_E_DATA;
739
740 ctx->ccix_host_info.ccix_host_mmap[ccix_mmap_idx].base =
741 ctx->config->mmap_table[i].base;
742 ctx->ccix_host_info.ccix_host_mmap[ccix_mmap_idx].size =
743 ctx->config->mmap_table[i].size;
744 ctx->ccix_host_info.ccix_host_mmap_count++;
745 }
746 }
747
748 FWK_LOG_INFO(MOD_NAME "Done");
749
750 ctx->initialized = true;
751
752 return FWK_SUCCESS;
753}
