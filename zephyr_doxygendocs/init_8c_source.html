1/**
2 * @file init.c
3 * @brief Zephyr 内核初始化模块
4 *
5 * 该模块是 Zephyr RTOS 内核的核心初始化入口，包含从 C 运行环境准备
6 * 完成后到用户应用 main() 函数执行之前的全部初始化流程。
7 *
8 * 主要函数及调用关系：
9 * z_cstart() -- 由 prep_c 调用，内核 C 入口
10 * ├── z_sys_init_run_level(EARLY)
11 * ├── arch_kernel_init()
12 * ├── z_device_state_init()
13 * ├── soc_early_init_hook() / board_early_init_hook()
14 * ├── z_sys_init_run_level(PRE_KERNEL_1)
15 * ├── z_sys_init_run_level(PRE_KERNEL_2)
16 * └── switch_to_main_thread(prepare_multithreading())
17 * └── bg_thread_main() -- main 线程入口
18 * ├── z_mem_manage_init() [MMU]
19 * ├── z_sys_init_run_level(POST_KERNEL)
20 * ├── soc_late_init_hook() / board_late_init_hook()
21 * ├── z_init_static_threads()
22 * ├── z_smp_init() [SMP]
23 * └── main()
24 *
25 * 初始化级别（init_level）按以下顺序依次执行：
26 * EARLY → PRE_KERNEL_1 → PRE_KERNEL_2 → POST_KERNEL → APPLICATION → [SMP]
27 *
28 * @note PRE_KERNEL_1/2 在调度器启动前执行，不可使用阻塞 API。
29 * @note POST_KERNEL 及之后的级别在调度器运行后执行。
30 *
31 * @copyright Copyright (c) 2010-2014 Wind River Systems, Inc.
32 * @license SPDX-License-Identifier: Apache-2.0
33 */

97/**
98 * @brief 初始化所有静态定义的线程
99 *
100 * 遍历 linker section 中所有 _static_thread_data 条目，调用 z_setup_new_thread()
101 * 完成线程创建。在 userspace 模式下，还会处理 k_object_assignment 条目，
102 * 将指定内核对象授权给对应线程。
103 *
104 * 静态线程可能设置了启动延迟（K_FOREVER 表示延迟启动，即 legacy 线程）。
105 * 非延迟线程会立即被调度执行。为防止线程在遍历过程中被抢占执行，
106 * 整个启动过程在调度器锁定期间完成。
107 */
108static void z_init_static_threads(void)
109{
110 STRUCT_SECTION_FOREACH(_static_thread_data, thread_data) {
111 z_setup_new_thread(
112 thread_data->init_thread,
113 thread_data->init_stack,
114 thread_data->init_stack_size,
115 thread_data->init_entry,
116 thread_data->init_p1,
117 thread_data->init_p2,
118 thread_data->init_p3,
119 thread_data->init_prio,
120 thread_data->init_options,
121 thread_data->init_name);
122
123 thread_data->init_thread->init_data = thread_data;
124 }
125
126#ifdef CONFIG_USERSPACE
127 /* 处理 userspace 对象授权：将内核对象访问权限授予指定线程 */
128 STRUCT_SECTION_FOREACH(k_object_assignment, pos) {
129 for (int i = 0; pos->objects[i] != NULL; i++) {
130 k_object_access_grant(pos->objects[i],
131 pos->thread);
132 }
133 }
134#endif /* CONFIG_USERSPACE */
135
136 /*
137 * 非延迟静态线程可以立即启动或按指定延迟启动。
138 * 即使调度器已锁定，tick 仍可递增和处理。
139 * 加锁以防止线程在全部注册完成前被调度执行。
140 *
141 * 注意：使用 legacy API 定义的静态线程延迟为 K_FOREVER，
142 * 即不会在此处启动。
143 */
144 k_sched_lock();
145 STRUCT_SECTION_FOREACH(_static_thread_data, thread_data) {
146 k_timeout_t init_delay = Z_THREAD_INIT_DELAY(thread_data);
147
148 if (!K_TIMEOUT_EQ(init_delay, K_FOREVER)) {
149 thread_schedule_new(thread_data->init_thread,
150 init_delay);
151 }
152 }
153 k_sched_unlock();
154}
155#else
156#define z_init_static_threads() do { } while (false)
157#endif /* CONFIG_MULTITHREADING */

279/**
280 * @brief 执行指定初始化级别的所有初始化条目
281 *
282 * 遍历由链接脚本按优先级排序的 init_entry 数组，依次调用每个
283 * 条目的初始化函数。每个条目可能关联一个 device（通过 dev 字段），
284 * 或仅包含一个 init_fn 纯函数。
285 *
286 * 对于关联了 device 的条目，若设备设置了 DEVICE_FLAG_INIT_DEFERRED
287 * 标志，则跳过在此处初始化（通常由驱动模型在稍后阶段处理）。
288 *
289 * @param[in] level 要执行的初始化级别
290 *
291 * @see INIT_ENTRY_DEFINE() 注册初始化条目的宏
292 * @see SYS_INIT() 便捷的设备初始化注册宏
293 */
294static void z_sys_init_run_level(enum init_level level)
295{
296 /** 各级别在 linker section 中的起始位置 */
297 static const struct init_entry *levels[] = {
298 __init_EARLY_start,
299 __init_PRE_KERNEL_1_start,
300 __init_PRE_KERNEL_2_start,
301 __init_POST_KERNEL_start,
302 __init_APPLICATION_start,
303#ifdef CONFIG_SMP
304 __init_SMP_start,
305#endif /* CONFIG_SMP */
306 /* 结束标记 */
307 __init_end,
308 };
309 const struct init_entry *entry;
310
311 for (entry = levels[level]; entry < levels[level+1]; entry++) {
312 const struct device *dev = entry->dev;
313 int result = 0;
314
315 sys_trace_sys_init_enter(entry, level);
316 if (dev != NULL) {
317 if ((dev->flags & DEVICE_FLAG_INIT_DEFERRED) == 0U) {
318 result = do_device_init(dev);
319 }
320 } else {
321 result = entry->init_fn();
322 }
323 sys_trace_sys_init_exit(entry, level, result);
324 }
325}

355/**
356 * @brief 内核 main 线程入口函数
357 *
358 * 这是 Zephyr 内核 main 线程的主体函数，在调度器启动后执行。
359 * 按顺序完成以下工作：
360 * 1. 内存管理初始化（MMU 模式下）
361 * 2. 设置 POST_KERNEL 完成标志
362 * 3. 执行 POST_KERNEL 级别初始化
363 * 4. 调用 SoC/Board 延迟初始化 hook
364 * 5. 启用栈指针随机化
365 * 6. 执行 APPLICATION 级别初始化
366 * 7. 初始化并启动所有静态定义的线程
367 * 8. SMP 多核初始化（若启用）
368 * 9. 调用用户应用 main() 函数
369 * 10. main() 返回后标记线程为非核心线程
370 * 11. 导出覆盖率数据（若启用）
371 *
372 * @param[in] unused1 未使用参数
373 * @param[in] unused2 未使用参数
374 * @param[in] unused3 未使用参数
375 */
376__boot_func
377static void bg_thread_main(void *unused1, void *unused2, void *unused3)
378{
379 ARG_UNUSED(unused1);
380 ARG_UNUSED(unused2);
381 ARG_UNUSED(unused3);
382
383#ifdef CONFIG_MMU
384 /*
385 * 内存管理子系统在此初始化，以便 backing store 或驱逐算法
386 * 可以初始化内核对象，同时允许 POST_KERNEL 及之后的所有任务
387 * 执行内存管理操作（k_mem_map_phys_bare() 除外，它随时可用）。
388 */
389 z_mem_manage_init();
390#endif /* CONFIG_MMU */
391 z_sys_post_kernel = true;
392
393#if CONFIG_IRQ_OFFLOAD
394 arch_irq_offload_init();
395#endif
396 z_sys_init_run_level(INIT_LEVEL_POST_KERNEL);
397
398 soc_late_init_hook();
399
400 board_late_init_hook();
401
402#if defined(CONFIG_STACK_POINTER_RANDOM) && (CONFIG_STACK_POINTER_RANDOM != 0)
403 z_stack_adjust_initialized = 1;
404#endif /* CONFIG_STACK_POINTER_RANDOM */
405
406#ifdef CONFIG_STATIC_INIT_GNU
407 z_static_init_gnu();
408#endif /* CONFIG_STATIC_INIT_GNU */
409
410 /* 应用层初始化（用户 main() 之前的最后一个 init level） */
411 z_sys_init_run_level(INIT_LEVEL_APPLICATION);
412
413 z_init_static_threads();
414
415#ifdef CONFIG_KERNEL_COHERENCE
416 __ASSERT_NO_MSG(sys_cache_is_mem_coherent(&_kernel));
417#endif /* CONFIG_KERNEL_COHERENCE */
418
419#ifdef CONFIG_SMP
420 if (!IS_ENABLED(CONFIG_SMP_BOOT_DELAY)) {
421 z_smp_init();
422 }
423 z_sys_init_run_level(INIT_LEVEL_SMP);
424#endif /* CONFIG_SMP */
425
426#ifdef CONFIG_MMU
427 z_mem_manage_boot_finish();
428#endif /* CONFIG_MMU */
429
430#ifdef CONFIG_BOOTARGS
431 extern int main(int, char **);
432 extern char **sys_boot_prepare_main_args(int *argc);
433
434 int argc = 0;
435 char **argv = sys_boot_prepare_main_args(&argc);
436 (void)main(argc, argv);
437#else
438 extern int main(void);
439
440 (void)main();
441#endif /* CONFIG_BOOTARGS */
442
443 /* main() 已完成，标记 main 线程为非核心线程 */
444 z_thread_essential_clear(&z_main_thread);
445
446#ifdef CONFIG_COVERAGE_DUMP
447 /* main() 退出后导出覆盖率数据 */
448 gcov_coverage_dump();
449#elif defined(CONFIG_COVERAGE_SEMIHOST)
450 gcov_coverage_semihost();
451#endif /* CONFIG_COVERAGE_DUMP */
452} /* LCOV_EXCL_LINE ... 覆盖率数据已在上方导出 */
453
454#if defined(CONFIG_MULTITHREADING)
455/**
456 * @brief 创建指定 CPU 的 idle 线程
457 *
458 * 为 CPU i 创建 idle 线程，以最低优先级（K_IDLE_PRIO）运行，
459 * 标记为核心线程（K_ESSENTIAL），线程入口为 idle() 函数。
460 *
461 * @param[in] i CPU 编号
462 */
463__boot_func
464static void init_idle_thread(int i)
465{
466 struct k_thread *thread = &z_idle_threads[i];
467 k_thread_stack_t *stack = z_idle_stacks[i];
468 size_t stack_size = K_KERNEL_STACK_SIZEOF(z_idle_stacks[i]);
469
470#ifdef CONFIG_THREAD_NAME
471
472#if CONFIG_MP_MAX_NUM_CPUS > 1
473 char tname[8];
474 snprintk(tname, 8, "idle %02d", i);
475#else
476 char *tname = "idle";
477#endif /* CONFIG_MP_MAX_NUM_CPUS */
478
479#else
480 char *tname = NULL;
481#endif /* CONFIG_THREAD_NAME */
482
483 z_setup_new_thread(thread, stack,
484 stack_size, idle, &_kernel.cpus[i],
485 NULL, NULL, K_IDLE_PRIO, K_ESSENTIAL,
486 tname);
487 z_mark_thread_as_not_sleeping(thread);
488
489#ifdef CONFIG_SMP
490 thread->base.is_idle = 1U;
491#endif /* CONFIG_SMP */
492}

494/**
495 * @brief 初始化指定 CPU 的内核数据
496 *
497 * 完成单个 CPU 的 idle 线程创建、中断栈设置、CPU ID 分配、
498 * 电源管理计数更新、对象核心系统注册和 IPI 工作队列初始化。
499 *
500 * @param[in] id CPU 编号
501 */
502void z_init_cpu(int id)
503{
504 init_idle_thread(id);
505 _kernel.cpus[id].idle_thread = &z_idle_threads[id];
506 _kernel.cpus[id].id = id;
507 _kernel.cpus[id].irq_stack =
508 (K_KERNEL_STACK_BUFFER(z_interrupt_stacks[id]) +
509 K_KERNEL_STACK_SIZEOF(z_interrupt_stacks[id]));
510#ifdef CONFIG_SCHED_THREAD_USAGE_ALL
511 _kernel.cpus[id].usage = &_kernel.usage[id];
512 _kernel.cpus[id].usage->track_usage =
513 CONFIG_SCHED_THREAD_USAGE_AUTO_ENABLE;
514#endif
515
516#ifdef CONFIG_PM
517 /*
518 * 递增活跃 CPU 计数。电源管理子系统从此处开始跟踪。
519 */
520 atomic_inc(&_cpus_active);
521#endif
522
523#ifdef CONFIG_OBJ_CORE_SYSTEM
524 k_obj_core_init_and_link(K_OBJ_CORE(&_kernel.cpus[id]), &obj_type_cpu);
525#ifdef CONFIG_OBJ_CORE_STATS_SYSTEM
526 k_obj_core_stats_register(K_OBJ_CORE(&_kernel.cpus[id]),
527 _kernel.cpus[id].usage,
528 sizeof(struct k_cycle_stats));
529#endif
530#endif
531
532#ifdef CONFIG_SCHED_IPI_SUPPORTED
533 sys_dlist_init(&_kernel.cpus[id].ipi_workq);
534#endif
535}

537/**
538 * @brief 准备多线程运行环境
539 *
540 * 初始化内核调度器、创建 main 线程、初始化 CPU 0 的 idle 线程
541 * 及其中断栈。在单核模式下，预先将 main 线程加载到调度器优先级缓存中，
542 * 因为它将是第一个运行的线程。
543 *
544 * @return main 线程的初始栈指针，用于上下文切换
545 *
546 * @note 调用此函数时 _kernel 结构体的所有字段为零（.bss 已清零），
547 * 对许多字段而言零初始化已足够。
548 */
549__boot_func
550static char *prepare_multithreading(void)
551{
552 char *stack_ptr;
553
554 /* _kernel.ready_q 已全部清零 */
555 z_sched_init();
556
557#ifndef CONFIG_SMP
558 /*
559 * 预先在调度器缓存中加载 main 线程，因为：
560 * - 调度器缓存不允许为 NULL
561 * - main 线程将第一个被调度执行
562 * - 此时无其他线程初始化，其优先级字段含有垃圾值，
563 * 会干扰缓存加载算法
564 */
565 _kernel.ready_q.cache = &z_main_thread;
566#endif /* CONFIG_SMP */
567 stack_ptr = z_setup_new_thread(&z_main_thread, z_main_stack,
568 K_THREAD_STACK_SIZEOF(z_main_stack),
569 bg_thread_main,
570 NULL, NULL, NULL,
571 CONFIG_MAIN_THREAD_PRIORITY,
572 K_ESSENTIAL, "main");
573 z_mark_thread_as_not_sleeping(&z_main_thread);
574 z_ready_thread(&z_main_thread);
575
576 z_init_cpu(0);
577
578 return stack_ptr;
579}

607/**
608 * @brief 获取早期启动阶段的伪随机数
609 *
610 * 在内核完整初始化之前提供伪随机数据，用于栈保护金丝雀值等
611 * 需要在启动早期获取随机数据的场景。
612 *
613 * 优先尝试使用硬件 entropy 驱动（ISR 模式），若驱动不可用
614 * 或返回数据不足，则回退到基于 cycle counter 的线性同余
615 * 伪随机数生成器（LCG）。
616 *
617 * @param[out] buf 输出缓冲区
618 * @param[in] length 需要生成的随机数据字节数
619 *
620 * @note 此函数使用固定种子的 LCG，安全性不适用于加密用途。
621 * @note 函数标记为 __weak，可被平台代码覆盖以使用硬件 TRNG。
622 */
623__boot_func
624void __weak z_early_rand_get(uint8_t *buf, size_t length)
625{
626 static uint64_t state = (uint64_t)CONFIG_TIMER_RANDOM_INITIAL_STATE;
627 int rc;
628
629#ifdef CONFIG_ENTROPY_HAS_DRIVER
630 const struct device *const entropy = DEVICE_DT_GET_OR_NULL(DT_CHOSEN(zephyr_entropy));
631
632 if ((entropy != NULL) && device_is_ready(entropy)) {
633 /* 尝试使用驱动提供的 ISR 专用 API 获取真随机数 */
634 rc = entropy_get_entropy_isr(entropy, buf, length, ENTROPY_BUSYWAIT);
635 if (rc > 0) {
636 length -= rc;
637 buf += rc;
638 }
639 }
640#endif /* CONFIG_ENTROPY_HAS_DRIVER */
641
642 /* 回退到基于 cycle counter 的 LCG 伪随机数生成 */
643 while (length > 0) {
644 uint32_t val;
645
646 state = state + k_cycle_get_32();
647 state = state * 2862933555777941757ULL + 3037000493ULL;
648 val = (uint32_t)(state >> 32);
649 rc = min(length, sizeof(val));
650 arch_early_memcpy((void *)buf, &val, rc);
651
652 length -= rc;
653 buf += rc;
654 }
655}

657/**
658 * @brief Zephyr 内核 C 入口函数
659 *
660 * 这是内核从汇编启动代码（经 prep_c）进入后的第一个 C 函数。
661 * 处理器必须处于正确的运行模式（32 位），且 .bss 段必须已清零。
662 *
663 * 按以下顺序执行初始化：

665 * 2. EARLY 级别系统初始化
666 * 3. 架构相关内核初始化
667 * 4. 日志核心初始化
668 * 5. dummy 线程初始化（多线程模式）
669 * 6. 静态设备状态初始化
670 * 7. SoC/Board 早期初始化 hook

672 * 9. SMP 架构初始化（若启用）
673 * 10. PRE_KERNEL_2 级别初始化
674 * 11. 栈保护金丝雀值初始化
675 * 12. timing 功能初始化（若启用）
676 * 13. 切换到 main 线程（多线程）或直接调用 bg_thread_main（单线程）
677 *
678 * @note 此函数不会返回（FUNC_NORETURN）。
679 * @note 标记为 FUNC_NO_STACK_PROTECTOR，因为在栈保护金丝雀值
680 * 初始化之前调用，此时 __stack_chk_guard 尚未设置。
681 */
682__boot_func
683FUNC_NO_STACK_PROTECTOR

691

693 arch_kernel_init();
694
695 LOG_CORE_INIT();
696
697#if defined(CONFIG_MULTITHREADING)
698 z_dummy_thread_init(&_thread_dummy);
699#endif /* CONFIG_MULTITHREADING */
700 /* 初始化所有静态设备的状态对象 */
701 z_device_state_init();
702
703 soc_early_init_hook();
704
705 board_early_init_hook();
706
707 /* 基础硬件初始化 */
708 z_sys_init_run_level(INIT_LEVEL_PRE_KERNEL_1);
709#if defined(CONFIG_SMP)
710 arch_smp_init();
711#endif
712 z_sys_init_run_level(INIT_LEVEL_PRE_KERNEL_2);
713
714#ifdef CONFIG_REQUIRES_STACK_CANARIES
715 uintptr_t stack_guard;
716
717 z_early_rand_get((uint8_t *)&stack_guard, sizeof(stack_guard));
718 __stack_chk_guard = stack_guard;
719 __stack_chk_guard <<= 8;
720#endif /* CONFIG_REQUIRES_STACK_CANARIES */
721
722#ifdef CONFIG_TIMING_FUNCTIONS_NEED_AT_BOOT
723 timing_init();
724 timing_start();
725#endif /* CONFIG_TIMING_FUNCTIONS_NEED_AT_BOOT */
726
727#ifdef CONFIG_MULTITHREADING
728 switch_to_main_thread(prepare_multithreading());
729#else
730#ifdef ARCH_SWITCH_TO_MAIN_NO_MULTITHREADING
731 /* 自定义架构例程：在无多线程模式下切换到 main() */
732 ARCH_SWITCH_TO_MAIN_NO_MULTITHREADING(bg_thread_main,
733 NULL, NULL, NULL);
734#else
735 bg_thread_main(NULL, NULL, NULL);
736
737 /* LCOV_EXCL_START
738 * 覆盖率数据已在上方导出
739 */
740 irq_lock();
741 while (true) {
742 }
743 /* LCOV_EXCL_STOP */
744#endif /* ARCH_SWITCH_TO_MAIN_NO_MULTITHREADING */
745#endif /* CONFIG_MULTITHREADING */
746
747 /*
748 * 编译器无法推断上述函数不会返回，需要显式标记
749 * 以消除警告。
750 */
751
752 CODE_UNREACHABLE; /* LCOV_EXCL_LINE */
753}
