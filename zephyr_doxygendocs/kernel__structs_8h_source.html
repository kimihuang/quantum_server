115/*
116 * ============================================================
117 * 线程优先级队列抽象
118 * ============================================================
119 *
120 * Zephyr 提供三种优先级队列实现，分别适用于不同场景：
121 *
122 * 1. CONFIG_SCHED_SIMPLE（简单双向链表）：
123 * - 数据结构：sys_dlist_t（排序双向链表）
124 * - 插入复杂度：O(N)
125 * - 代码体积：最小
126 * - 适用场景：线程数量少、对代码体积敏感的系统
127 *
128 * 2. CONFIG_SCHED_SCALABLE（可扩展红黑树）：
129 * - 数据结构：rbtree（平衡二叉搜索树）
130 * - 插入复杂度：O(logN)
131 * - 代码体积：较大
132 * - 适用场景：线程数量多、需要 O(logN) 调度性能的系统
133 *
134 * 3. CONFIG_SCHED_MULTIQ（多队列）：
135 * - 数据结构：每个优先级一个独立链表 + 位图索引
136 * - 插入复杂度：O(1)
137 * - 代码体积：RAM 占用较大（每个优先级一个链表头）
138 * - 适用场景：优先级数量有限、需要极快调度响应的系统
139 * - 限制：不适用于 deadline scheduling 等需要大优先级空间的特性
140 *
141 * 等待队列（wait_q）可选用简单链表或红黑树。
142 * 系统就绪队列（ready_q）可选用三种中任意一种。
143 * 均在构建时通过 Kconfig 配置选择。
144 */
145
146/**
147 * @brief 基于红黑树的优先级队列
148 *
149 * @details 使用平衡二叉搜索树（rbtree）实现可扩展的优先级队列。
150 * 线程按优先级排序存储，支持 O(logN) 的插入和删除。
151 * next_order_key 用于在同优先级线程之间维护 FIFO 顺序。
152 */
153struct _priq_rb {
154 struct rbtree tree; /**< 红黑树根节点 */
155 int next_order_key; /**< 下一个排序键，同优先级线程按此值排序保证 FIFO */
156};
157
158/**
159 * @brief 基于多队列的优先级队列
160 *
161 * @details 为每个优先级维护一个独立的双向链表，通过位图（bitmask）
162 * 快速定位非空链表中的最高优先级，实现 O(1) 的最高优先级查找。
163 */
164struct _priq_mq {
165 sys_dlist_t queues[K_NUM_THREAD_PRIO]; /**< 每个优先级一个链表 */
166 unsigned long bitmask[PRIQ_BITMAP_SIZE]; /**< 位图：bit[i]=1 表示优先级 i 有就绪线程 */
167#ifndef CONFIG_SMP
168 unsigned int cached_queue_index; /**< 缓存的最高优先级索引（单核优化，避免重复位图扫描） */
169#endif
170};

172/**
173 * @brief 系统就绪队列
174 *
175 * @details 调度器的核心数据结构，存放所有处于就绪状态（可被调度）的线程。
176 * 根据配置使用不同的底层实现（简单链表/红黑树/多队列）。
177 */
178struct _ready_q {
179#ifndef CONFIG_SMP
180 /**
181 * @brief 调度缓存：指向下一个应被执行的线程
182 *
183 * @details 单核模式下始终包含下一个要运行的线程，不允许为 NULL。
184 * 初始化时设为 main 线程，避免空指针检查开销。
185 */
186 struct k_thread *cache;
187#endif
188
189#if defined(CONFIG_SCHED_SIMPLE)
190 sys_dlist_t runq; /**< 简单双向链表：按优先级排序的线程链表 */
191#elif defined(CONFIG_SCHED_SCALABLE)
192 struct _priq_rb runq; /**< 红黑树：可扩展的优先级队列 */
193#elif defined(CONFIG_SCHED_MULTIQ)
194 struct _priq_mq runq; /**< 多队列：每个优先级独立链表 + 位图索引 */
195#endif
196};

308/**
309 * @brief Zephyr 全局内核数据结构（唯一实例 _kernel）
310 *
311 * @details 这是 Zephyr 内核的全局状态根结构体，在 init.c 中定义。
